package br.gov.ba.saj.smtt.cadmototaxistas.model.enums;

import lombok.AllArgsConstructor;
import lombok.Getter;

/**
 * Enumeração que representa as modalidades de atuação de um condutor
 * habilitado.
 * <p>
 * Define se o condutor atua como mototaxista, motofretista ou exerce ambas as
 * funções.
 * </p>
 *
 * <ul>
 * <li>{@code MOTOTAXISTA} - Transporte individual de passageiros em
 * motocicleta.</li>
 * <li>{@code MOTOFRETISTA} - Transporte de pequenas cargas ou encomendas em
 * motocicleta.</li>
 * <li>{@code AMBAS} - Atua tanto como mototaxista quanto como
 * motofretista.</li>
 * </ul>
 *
 * Esta enum é utilizada para atribuir e validar o tipo de serviço prestado pelo
 * condutor.
 */
@Getter
@AllArgsConstructor
public enum EModalidadeDeCondutor {

    /** Condutor que presta serviço de transporte de passageiros. */
    MOTOTAXISTA("MOTOTAXISTA", "M"),

    /** Condutor que presta serviço de transporte de cargas leves. */
    MOTOFRETISTA("MOTOFRETISTA", "F"),

    /** Condutor que exerce ambas as funções: transporte de passageiros e cargas. */
    AMBAS("AMBAS", "A");

    /**
     * Representação textual da modalidade do condutor.
     */
    private final String modalidadeDeCondutor;

    /**
     * Prefixo utilizado para identificar a modalidade do condutor.
     * Pode ser usado em sistemas de codificação ou abreviação.
     */
    private final String prefixo;
}
