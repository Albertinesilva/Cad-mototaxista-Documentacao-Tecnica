package br.gov.ba.saj.smtt.cadmototaxistas.model.enums;

import lombok.AllArgsConstructor;
import lombok.Getter;

/**
 * Enumeração que representa a situação da posse da motocicleta utilizada pelo condutor.
 * <p>
 * Indica se a moto é de propriedade do condutor ou se é locada.
 * </p>
 * 
 * <ul>
 *   <li>{@code PROPRIA} - A motocicleta pertence ao próprio condutor.</li>
 *   <li>{@code LOCADA} - A motocicleta é alugada de terceiros.</li>
 * </ul>
 * 
 * Esta enum é utilizada para classificar a situação da moto no cadastro do sistema.
 */
@Getter
@AllArgsConstructor
public enum ESituacaoMoto {

    /** Motocicleta de propriedade do condutor. */
    PROPRIA("Moto Própria"),

    /** Motocicleta alugada (locada) para uso profissional. */
    LOCADA("Moto Locada");

    /**
     * Descrição textual da situação da motocicleta.
     */
    private final String ePropria;
}
