package br.gov.ba.saj.smtt.cadmototaxistas.model.enums;

import lombok.AllArgsConstructor;
import lombok.Getter;

/**
 * Enumeração que representa as categorias da Carteira Nacional de Habilitação
 * (CNH).
 * <p>
 * Cada valor representa uma combinação de permissões para condução de veículos
 * conforme as regras do DETRAN.
 * </p>
 * 
 * <ul>
 * <li>{@code A} - Veículos de duas ou três rodas (motos).</li>
 * <li>{@code AB} - Veículos de duas ou três rodas e veículos de quatro rodas
 * até 3.500 kg.</li>
 * <li>{@code AC} - Categoria A + veículos de carga acima de 3.500 kg.</li>
 * <li>{@code AD} - Categoria B + veículos para transporte de passageiros.</li>
 * <li>{@code AE} - Categoria B + combinação de unidades acopladas acima de
 * 6.000 kg.</li>
 * </ul>
 * 
 * Esta enum é utilizada para definir e restringir os valores possíveis no campo
 * de categoria da CNH.
 */
@Getter
@AllArgsConstructor
public enum ECategoria {

    /** Categoria A: motos, ciclomotores, motonetas e triciclos. */
    A("A"),

    /** Categoria AB: motos e veículos de até 3.500 kg (categoria B). */
    AB("AB"),

    /** Categoria AC: motos e veículos de carga acima de 3.500 kg (categoria C). */
    AC("AC"),

    /**
     * Categoria AD: motos e veículos de transporte de passageiros (categoria D).
     */
    AD("AD"),

    /** Categoria AE: motos e veículos com unidade acoplada (categoria E). */
    AE("AE");

    /**
     * Representação textual da categoria.
     */
    private final String categoria;

}
