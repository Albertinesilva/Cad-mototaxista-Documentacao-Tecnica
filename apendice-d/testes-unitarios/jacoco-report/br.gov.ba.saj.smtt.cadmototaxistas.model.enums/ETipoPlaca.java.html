package br.gov.ba.saj.smtt.cadmototaxistas.model.enums;

import lombok.AllArgsConstructor;
import lombok.Getter;

/**
 * Enumeração que representa os tipos de placa veicular reconhecidos pelo
 * sistema.
 * <p>
 * Essa enum é usada para identificar o modelo da placa da motocicleta
 * cadastrada.
 * </p>
 *
 * <ul>
 * <li>{@code MERCOSUL} - Placa no novo padrão Mercosul (branca com letras
 * azuis).</li>
 * <li>{@code ANTIGA} - Placa no modelo antigo (geralmente cinza ou preta).</li>
 * </ul>
 */
@Getter
@AllArgsConstructor
public enum ETipoPlaca {

    /** Placa padrão Mercosul (nova). */
    MERCOSUL("Mercosul"),

    /** Placa antiga (cinza ou preta). */
    ANTIGA("Cinza");

    /**
     * Descrição textual do tipo de placa.
     */
    private final String tipoPlaca;
}
