package br.gov.ba.saj.smtt.cadmototaxistas.model.enums;

import lombok.AllArgsConstructor;
import lombok.Getter;

import java.util.Arrays;

/**
 * Enumeração que representa os diferentes tipos sanguíneos reconhecidos pelo
 * sistema.
 * <p>
 * Os valores estão formatados no padrão brasileiro (A+, O-, etc).
 * </p>
 *
 * <p>
 * Exemplos de uso:
 * </p>
 * 
 * <pre>
 * ETipoSanguineo tipo = ETipoSanguineo.fromValue("O+");
 * String descricao = tipo.getTipoSanguineo(); // "O+"
 * </pre>
 */
@Getter
@AllArgsConstructor
public enum ETipoSanguineo {

    /** Tipo sanguíneo A positivo. */
    APOSITIVO("A+"),

    /** Tipo sanguíneo B positivo. */
    BPOSITIVO("B+"),

    /** Tipo sanguíneo AB positivo. */
    ABPOSITIVO("AB+"),

    /** Tipo sanguíneo O positivo. */
    OPOSITIVO("O+"),

    /** Tipo sanguíneo A negativo. */
    ANEGATIVO("A-"),

    /** Tipo sanguíneo B negativo. */
    BNEGATIVO("B-"),

    /** Tipo sanguíneo AB negativo. */
    ABNEGATIVO("AB-"),

    /** Tipo sanguíneo O negativo. */
    ONEGATIVO("O-");

    /**
     * Representação textual do tipo sanguíneo.
     */
    private final String tipoSanguineo;

    /**
     * Obtém a enum {@link ETipoSanguineo} correspondente ao valor informado.
     *
     * @param value o valor textual do tipo sanguíneo (ex: "A+", "O-").
     * @return o enum correspondente ao valor.
     * @throws IllegalArgumentException se o valor não corresponder a nenhum tipo
     *                                  conhecido.
     */
    public static ETipoSanguineo fromValue(String value) {
        return Arrays.stream(values())
                .filter(t -> t.getTipoSanguineo().equalsIgnoreCase(value))
                .findFirst()
                .orElseThrow(() -> new IllegalArgumentException("Tipo sanguíneo inválido: " + value));
    }
}
