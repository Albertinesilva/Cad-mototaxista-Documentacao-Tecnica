package br.gov.ba.saj.smtt.cadmototaxistas.auditoria.controller;

import java.util.List;
import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.ModelMap;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.servlet.mvc.support.RedirectAttributes;

import br.gov.ba.saj.smtt.cadmototaxistas.auditoria.model.entities.RegistroAuditoria;
import br.gov.ba.saj.smtt.cadmototaxistas.auditoria.service.RegistroAuditoria.RegistroAuditoriaService;
import br.gov.ba.saj.smtt.cadmototaxistas.util.paginacao.PaginacaoUtil;

/**
 * Controller responsável por lidar com requisições relacionadas à exibição dos
 * registros de auditoria.
 *
 * Esta classe expõe endpoints para visualização das ações realizadas no
 * sistema,
 * registradas na tabela de auditoria.
 * 
 * <p>
 * Os dados são utilizados principalmente para fins de rastreabilidade e análise
 * de ações dos usuários.
 * </p>
 *
 * Mapeada sob o caminho base <code>/auditorias/registro</code>.
 * 
 * @author SeuNome
 * @since 1.0
 */
@Controller
@RequestMapping("/auditorias")
public class RegistroAuditoriaController {

  /**
   * Serviço responsável pela lógica de negócio relacionada aos registros de
   * auditoria.
   * Injetado automaticamente pelo Spring.
   */
  @Autowired
  private RegistroAuditoriaService registroAuditoriaService;

  /**
   * Exibe a lista paginada e ordenada das auditorias de registros.
   * 
   * <p>
   * Este método é responsável por recuperar uma página da listagem de
   * {@link RegistroAuditoria},
   * com base nos parâmetros fornecidos via requisição (número da página e direção
   * da ordenação),
   * e adicioná-la ao modelo da view para exibição em tela.
   * </p>
   *
   * @param model o {@link ModelMap} utilizado para adicionar os dados à view.
   * @param page  o número da página atual (opcional; se ausente, assume 1).
   * @param dir   a direção da ordenação ("asc" para ascendente ou "desc" para
   *              descendente; padrão é "asc").
   * @return o nome da view que renderiza a lista de registros de auditoria,
   *         com suporte a paginação.
   */
  @GetMapping("/registros/listar")
  public String listar(ModelMap model, @RequestParam("page") Optional<Integer> page,
      @RequestParam("dir") Optional<String> dir) {
    int paginaAtual = page.orElse(1);
    String ordem = dir.orElse("asc");

    PaginacaoUtil<RegistroAuditoria> pageAuditoria = registroAuditoriaService.buscaPaginada(paginaAtual, ordem);
    model.addAttribute("pageAuditoria", pageAuditoria);
    return "auditoria/lista";
  }

  /**
   * Busca registros de auditoria filtrando pelo nome do funcionário.
   * 
   * <p>
   * Este método permite pesquisar registros de auditoria com base no nome do
   * funcionário,
   * retornando uma lista de registros que contêm o nome informado.
   * </p>
   *
   * @param nome  o nome do funcionário a ser pesquisado (parcial ou completo).
   * @param model objeto {@link ModelMap} usado para passar atributos para a view.
   * @param attr  atributos para redirecionamento com mensagens de sucesso ou
   *              falha.
   * @return o nome da view que renderiza a lista de registros de auditoria, ou
   *         redireciona com mensagem de erro se nenhum registro for encontrado.
   */
  @GetMapping("/buscar/por-nome-de-funcionario")
  public String getPorNome(@RequestParam("nome") String nome, ModelMap model, RedirectAttributes attr) {
    List<RegistroAuditoria> registrosAuditoria = registroAuditoriaService.buscarPorNomeContendo(nome);

    if (registrosAuditoria.isEmpty()) {
      attr.addFlashAttribute("fail", "Nenhuma registro de auditoria encontrado com esse nome de funcionário.");
      return "redirect:/auditorias/registros/listar";
    }

    model.addAttribute("registrosAuditoria", registrosAuditoria);
    return "auditoria/lista";
  }
}
