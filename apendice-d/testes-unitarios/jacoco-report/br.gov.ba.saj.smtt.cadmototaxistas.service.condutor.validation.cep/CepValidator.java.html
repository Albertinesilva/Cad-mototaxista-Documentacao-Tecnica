package br.gov.ba.saj.smtt.cadmototaxistas.service.condutor.validation.cep;

import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;

/**
 * Validador para verificar se um valor de CEP (Código de Endereçamento Postal)
 * está no formato válido "xxxxx-xxx".
 * 
 * O formato esperado é 5 dígitos, seguido de hífen e mais 3 dígitos, por exemplo:
 * "12345-678".
 * 
 * Caso o valor seja null ou vazio, o validador considera válido.
 */
public class CepValidator implements ConstraintValidator<ValidCep, String> {

  private static final String CEP_REGEX = "^\\d{5}-\\d{3}$";

  /**
   * Valida se o valor informado está no formato correto de CEP.
   * 
   * @param value   Valor do CEP a ser validado.
   * @param context Contexto da validação.
   * @return true se o valor for null, vazio ou corresponder ao formato "xxxxx-xxx";
   *         false caso contrário.
   */
  @Override
  public boolean isValid(String value, ConstraintValidatorContext context) {
    if (value == null || value.trim().isEmpty()) {
      return true;
    }
    return value.matches(CEP_REGEX);
  }
}
