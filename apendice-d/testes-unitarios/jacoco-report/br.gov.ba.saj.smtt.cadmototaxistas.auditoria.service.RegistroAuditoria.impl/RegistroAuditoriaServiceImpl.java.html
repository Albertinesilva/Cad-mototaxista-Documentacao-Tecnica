package br.gov.ba.saj.smtt.cadmototaxistas.auditoria.service.RegistroAuditoria.impl;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.context.SecurityContextHolder;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Propagation;
import org.springframework.transaction.annotation.Transactional;

import br.gov.ba.saj.smtt.cadmototaxistas.auditoria.model.entities.RegistroAuditoria;
import br.gov.ba.saj.smtt.cadmototaxistas.auditoria.model.enums.AcaoAuditoria;
import br.gov.ba.saj.smtt.cadmototaxistas.auditoria.repository.RegistroAuditoriaRepository;
import br.gov.ba.saj.smtt.cadmototaxistas.auditoria.service.RegistroAuditoria.RegistroAuditoriaService;
import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Condutor;
import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Funcionario;
import br.gov.ba.saj.smtt.cadmototaxistas.repositories.FuncionarioRepository;
import br.gov.ba.saj.smtt.cadmototaxistas.util.paginacao.PaginacaoUtil;
import jakarta.persistence.EntityManager;
import jakarta.persistence.PersistenceContext;

/**
 * Implementação da interface {@link RegistroAuditoriaService}, responsável por
 * registrar eventos de auditoria relacionados a operações executadas sobre a
 * entidade {@link Condutor}.
 *
 * <p>
 * Esta classe integra-se com o Spring Security para capturar o usuário
 * autenticado no momento da operação, e persiste um registro contendo:
 * </p>
 * <ul>
 * <li>Usuário que executou a ação</li>
 * <li>Dados do funcionário (nome, e-mail)</li>
 * <li>Dados do condutor afetado</li>
 * <li>Tipo da ação executada (CREATE, UPDATE, DELETE, CONSULT)</li>
 * <li>Data do evento</li>
 * </ul>
 * 
 * @author SeuNome
 * @since 1.0
 */
@Service
public class RegistroAuditoriaServiceImpl implements RegistroAuditoriaService {

    /**
     * Repositório responsável por salvar os registros de auditoria no banco de
     * dados.
     */
    @Autowired
    private RegistroAuditoriaRepository registroAuditoriaRepository;

    /**
     * Repositório utilizado para buscar dados do funcionário logado, através do
     * e-mail do usuário autenticado.
     */
    @Autowired
    private FuncionarioRepository funcionarioRepository;

    @PersistenceContext
    private EntityManager entityManager;

    /**
     * {@inheritDoc}
     *
     * <p>
     * Cria e persiste um registro de auditoria com as informações da ação
     * realizada,
     * dados do condutor afetado e dados do servidor autenticado no sistema.
     * </p>
     * 
     * @param acao     tipo de ação executada (ex: {@code CREATE}, {@code UPDATE},
     *                 {@code DELETE}).
     * @param condutor condutor relacionado à ação executada.
     */
    @Override
    @Transactional(propagation = Propagation.REQUIRES_NEW)
    public void registrar(AcaoAuditoria acao, Condutor condutor) {
        Authentication auth = SecurityContextHolder.getContext().getAuthentication();
        String emailUsuario = (auth != null) ? auth.getName() : "Sistema";

        Optional<Funcionario> servidorOpt = funcionarioRepository.findByUsuarioEmail(emailUsuario);

        RegistroAuditoria auditoria = new RegistroAuditoria();
        auditoria.setUsuario(servidorOpt.map(Funcionario::getUsuario).orElse(null));
        auditoria.setNomeFuncionario(servidorOpt.map(Funcionario::getNome).orElse("Sistema"));
        auditoria.setEmailFuncionario(emailUsuario);
        auditoria.setAcao(acao.name()); // Ex: CREATE, UPDATE
        auditoria.setDescricaoAcao(acao.getDescricao()); // Ex: Criar, Editar
        auditoria.setCondutor(condutor.getNome());
        auditoria.setNumeroDeOrdem(condutor.getNumOrdem());
        auditoria.setDataEvento(LocalDateTime.now());

        registroAuditoriaRepository.save(auditoria);
    }

    /**
     * {@inheritDoc}
     *
     * <p>
     * Recupera todos os registros de auditoria, ordenados pela data do evento de
     * forma decrescente.
     * </p>
     *
     * @return lista de {@link RegistroAuditoria} ordenada pela data mais recente.
     */
    @Override
    @Transactional(readOnly = true)
    public List<RegistroAuditoria> buscarTodos() {
        return registroAuditoriaRepository.findAllByOrderByDataEventoDesc();
    }

    /**
     * {@inheritDoc}
     *
     * <p>
     * Busca registros de auditoria filtrando pelo nome do funcionário, ignorando
     * maiúsculas e minúsculas.
     * </p>
     *
     * @param nome nome do funcionário a ser pesquisado (parcial ou completo).
     * @return lista de registros de auditoria que contêm o nome do funcionário
     *         informado.
     */
    @Override
    @Transactional(readOnly = true)
    public List<RegistroAuditoria> buscarPorNomeContendo(String nome) {
        return registroAuditoriaRepository.findByNomeFuncionarioContainingIgnoreCase(nome.trim());
    }

    /**
     * Realiza a busca paginada das auditorias, com ordenação conforme a direção
     * 
     * <p>
     * Este método utiliza a API do JPA (`EntityManager`) para executar uma consulta
     * JPQL que retorna
     * uma lista de objetos {@link RegistroAuditoria} de forma paginada, ou seja,
     * limitada
     * ao número de registros
     * por página e deslocada de acordo com o número da página atual. A ordenação é
     * feita com base na auditoria
     * </p>
     * 
     * <p>
     * O tamanho fixo da página é definido como 5 registros. A página 1 representa
     * os primeiros 5 elementos.
     * A direção da ordenação pode ser "asc" (crescente) ou "desc" (decrescente).
     * </p>
     *
     * @param pagina  o número da página solicitada (baseado em 1, e não 0).
     * @param direcao a direção da ordenação, sendo "asc" para ascendente ou "desc"
     *                para descendente.
     * @return um objeto {@link PaginacaoUtil} contendo os dados da página atual
     *         (lista de auditorias, total de páginas, etc.).
     */
    @Override
    public PaginacaoUtil<RegistroAuditoria> buscaPaginada(int pagina, String direcao) {
        int tamanho = 5;
        int inicio = (pagina - 1) * tamanho; // Ex: página 1 começa do índice 0
        List<RegistroAuditoria> auditorias = entityManager
                .createQuery("select r from RegistroAuditoria r order by r.nomeFuncionario " + direcao,
                        RegistroAuditoria.class)
                .setFirstResult(inicio)
                .setMaxResults(tamanho)
                .getResultList();

        long totalRegistros = count();
        long totalDePaginas = (totalRegistros + (tamanho - 1)) / tamanho;

        return new PaginacaoUtil<>(tamanho, pagina, totalDePaginas, direcao, auditorias);
    }

    /**
     * Conta o número total de registros da entidade {@link RegistroAuditoria} no
     * banco de
     * dados.
     *
     * <p>
     * Utiliza uma consulta JPQL simples com a função de agregação
     * <code>count()</code> para retornar
     * o total de registros de auditoria cadastrados.
     * </p>
     *
     * @return o número total de registros da entidade {@link RegistroAuditoria}.
     */
    @Override
    public long count() {
        return entityManager
                .createQuery("select count(r) from RegistroAuditoria r", Long.class)
                .getSingleResult();
    }
}
