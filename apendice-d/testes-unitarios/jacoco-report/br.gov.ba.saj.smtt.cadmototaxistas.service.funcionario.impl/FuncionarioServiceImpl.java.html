package br.gov.ba.saj.smtt.cadmototaxistas.service.funcionario.impl;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Funcionario;
import br.gov.ba.saj.smtt.cadmototaxistas.repositories.FuncionarioRepository;
import br.gov.ba.saj.smtt.cadmototaxistas.service.funcionario.FuncionarioService;

/**
 * Implementação do serviço {@link FuncionarioService} que encapsula as regras de
 * negócio relacionadas aos funcionários da SMTT.
 * 
 * Esta classe utiliza o repositório {@link FuncionarioRepository} para realizar
 * operações de persistência e recuperação de dados da entidade {@link Funcionario}.
 * 
 * Todas as operações são transacionais e, quando possível, marcadas como somente leitura
 * para melhor desempenho e controle de transações.
 */
@Service
public class FuncionarioServiceImpl implements FuncionarioService {
    
    @Autowired
    private FuncionarioRepository funcionarioRepository;

    /**
     * Busca um funcionário pelo ID do usuário associado.
     *
     * @param id ID do usuário
     * @return Instância de {@link Funcionario} ou um novo objeto se não encontrado
     */
    @Override
    @Transactional(readOnly = true)
    public Funcionario buscarPorUsuarioId(Long id){
        return funcionarioRepository.findByUsuarioId(id).orElse(new Funcionario());
    }

    /**
     * Salva um novo funcionário no banco de dados.
     *
     * @param funcionario Funcionário a ser salvo
     */
    @Override
    @Transactional
    public void salvar(Funcionario funcionario) {
       funcionarioRepository.save(funcionario);
    }

    /**
     * Edita os dados de um funcionário existente.
     *
     * @param funcionario Funcionário com dados atualizados
     */
    @Override
    @Transactional
    public void editar(Funcionario funcionario) {
        funcionarioRepository.save(funcionario);
    }

    /**
     * Exclui um funcionário pelo ID.
     *
     * @param id ID do funcionário a ser removido
     */
    @Override
    @Transactional
    public void excluir(Long id) {
        funcionarioRepository.deleteById(id);
    }

    /**
     * Busca todos os funcionários cadastrados.
     *
     * @return Lista de funcionários
     */
    @Override
    @Transactional(readOnly = true)
    public List<Funcionario> buscarTodos() {
        return funcionarioRepository.findAll();
    }

    /**
     * Busca funcionários pelo nome.
     * 
     * @param nome Nome do funcionário
     * @return Lista de funcionários (implementação pendente)
     */
    @Override
    @Transactional(readOnly = true)
    public List<Funcionario> buscarPorNome(String nome) {
        return null; // Implementação futura
    }

    /**
     * Busca um funcionário pelo e-mail do usuário associado.
     *
     * @param email E-mail do usuário
     * @return Instância de {@link Funcionario} ou um novo objeto se não encontrado
     */
    @Override
    @Transactional(readOnly = true)
    public Funcionario buscarPorEmail(String email) {
        return funcionarioRepository.findByUsuarioEmail(email).orElse(new Funcionario());
    }
}
