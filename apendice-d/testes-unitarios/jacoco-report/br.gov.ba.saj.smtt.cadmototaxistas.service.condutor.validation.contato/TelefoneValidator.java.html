package br.gov.ba.saj.smtt.cadmototaxistas.service.condutor.validation.contato;

import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;

/**
 * Validador para a anotação {@link ValidTelefone}.
 * 
 * Verifica se um número de telefone informado possui exatamente 15 caracteres,
 * que é o formato esperado para números como: (99) 99999-9999.
 * 
 * Também permite valores vazios ou nulos, considerando que a validação
 * de obrigatoriedade (caso necessária) deve ser feita separadamente
 * com outras anotações como {@code @NotBlank} ou {@code @NotNull}.
 */
public class TelefoneValidator implements ConstraintValidator<ValidTelefone, String> {

  // Expressão regular que permite string vazia ou com exatamente 15 caracteres
  private static final String TELEFONE_REGEX = "^$|^.{15}$";

  /**
   * Executa a validação do número de telefone.
   * 
   * @param value   Valor do campo anotado.
   * @param context Contexto da validação.
   * @return {@code true} se o valor for válido (vazio, nulo ou com 15 caracteres), caso contrário {@code false}.
   */
  @Override
  public boolean isValid(String value, ConstraintValidatorContext context) {
    if (value == null || value.trim().isEmpty()) {
      return true;
    }
    return value.matches(TELEFONE_REGEX);
  }
}
