package br.gov.ba.saj.smtt.cadmototaxistas.service.dados.formulario;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.ui.Model;
import org.springframework.ui.ModelMap;

import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.MotoMarca;
import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.MotoModelo;
import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Ponto;
import br.gov.ba.saj.smtt.cadmototaxistas.model.enums.*;
import br.gov.ba.saj.smtt.cadmototaxistas.service.motocicleta.marca.MotoMarcaService;
import br.gov.ba.saj.smtt.cadmototaxistas.service.motocicleta.modelo.MotoModeloService;
import br.gov.ba.saj.smtt.cadmototaxistas.service.ponto.PontoService;

/**
 * Serviço responsável por fornecer dados auxiliares utilizados em formulários
 * da aplicação.
 */
@Service
public class DadosFormularioService {

  @Autowired
  private PontoService pontoService;

  @Autowired
  private MotoMarcaService motoMarcaService;

  @Autowired
  private MotoModeloService motoModeloService;

  /**
   * Carrega os atributos fixos e dinâmicos para popular formulários (ex.: marcas,
   * modelos, pontos).
   *
   * @param model o modelo da view onde os dados serão inseridos.
   */
  public void carregarAtributosGerais(Model model) {
    model.addAttribute("motocicletaMarcas", getMarcasDeMotocicletaAtivas());
    model.addAttribute("motocicletaModelos", getModelosDeMotocicletaAtivos());
    model.addAttribute("pontos", getPontosAtivos());
  }

  /**
   * Carrega os enums necessários para dropdowns e seletores nos formulários.
   *
   * @param model o modelo da view onde os dados serão inseridos.
   */
  public void carregarEnumsParaFormulario(Model model) {
    model.addAttribute("tiposSanguineos", getTiposSanguineos());
    model.addAttribute("categoriasCnh", getCategoriasCnh());
    model.addAttribute("tiposPlaca", getTiposPlaca());
    model.addAttribute("situacoesMoto", getSituacoesMoto());
    model.addAttribute("modalidadesDeCondutor", getModalidadesDeCondutor());
  }

  /**
   * Carrega os atributos fixos e dinâmicos para popular formulários (ex.: marcas,
   * modelos, pontos).
   *
   * @param model o modelo da view onde os dados serão inseridos.
   */
  public void carregarAtributosGerais(ModelMap model) {
    model.addAttribute("motocicletaMarcas", getMarcasDeMotocicletaAtivas());
    model.addAttribute("motocicletaModelos", getModelosDeMotocicletaAtivos());
    model.addAttribute("pontos", getPontosAtivos());
  }

  /**
   * Carrega os enums necessários para dropdowns e seletores nos formulários.
   *
   * @param model o modelo da view onde os dados serão inseridos.
   */
  public void carregarEnumsParaFormulario(ModelMap model) {
    model.addAttribute("tiposSanguineos", getTiposSanguineos());
    model.addAttribute("categoriasCnh", getCategoriasCnh());
    model.addAttribute("tiposPlaca", getTiposPlaca());
    model.addAttribute("situacoesMoto", getSituacoesMoto());
    model.addAttribute("modalidadesDeCondutor", getModalidadesDeCondutor());
  }

  public List<Ponto> getPontosAtivos() {
    return pontoService.findAllPontosAtivos();
  }

  public List<MotoModelo> getModelosDeMotocicletaAtivos() {
    return motoModeloService.buscarTodos()
        .stream()
        .filter(MotoModelo::isAtivo)
        .toList();
  }

  public List<MotoMarca> getMarcasDeMotocicletaAtivas() {
    return motoMarcaService.buscarTodos()
        .stream()
        .filter(MotoMarca::isAtivo)
        .toList();
  }

  public ETipoSanguineo[] getTiposSanguineos() {
    return ETipoSanguineo.values();
  }

  public ECategoria[] getCategoriasCnh() {
    return ECategoria.values();
  }

  public ETipoPlaca[] getTiposPlaca() {
    return ETipoPlaca.values();
  }

  public ESituacaoMoto[] getSituacoesMoto() {
    return ESituacaoMoto.values();
  }

  public EModalidadeDeCondutor[] getModalidadesDeCondutor() {
    return EModalidadeDeCondutor.values();
  }
}
