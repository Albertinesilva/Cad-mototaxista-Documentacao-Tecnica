package br.gov.ba.saj.smtt.cadmototaxistas.security.model.enums;

/**
 * Enum que representa os tipos de perfil disponíveis no sistema.
 * 
 * <p>
 * Cada tipo possui um código identificador e uma descrição, que são utilizados
 * para controle de acesso e associação de permissões aos usuários.
 * </p>
 * 
 * <ul>
 * <li>{@code ADMIN} - Perfil de administrador com permissões amplas.</li>
 * <li>{@code FUNCIONARIO} - Perfil de funcionário da SMTT.</li>
 * </ul>
 */
public enum PerfilTipo {

	/** Perfil de administrador do sistema */
	ADMIN(1, "ADMIN"),

	/** Perfil de funcionário */
	FUNCIONARIO(2, "FUNCIONARIO"),

	/** Perfil de cliente */
	CLIENTE(3, "CLIENTE");

	private final long cod;
	private final String descricao;

	private PerfilTipo(long cod, String descricao) {
		this.cod = cod;
		this.descricao = descricao;
	}

	/**
	 * Retorna o código numérico do perfil.
	 * 
	 * @return o código do perfil
	 */
	public long getCod() {
		return cod;
	}

	/**
	 * Retorna a descrição do perfil.
	 * 
	 * @return a descrição do perfil
	 */
	public String getDescricao() {
		return descricao;
	}
}
