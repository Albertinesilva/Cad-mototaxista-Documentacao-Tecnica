package br.gov.ba.saj.smtt.cadmototaxistas.security.model.enums;

/**
 * Enum que define os tipos de tokens utilizados no sistema, juntamente com o
 * comprimento em bytes de cada um.
 *
 * <p>
 * Esses tokens são utilizados em processos de ativação de conta, redefinição de
 * senha,
 * verificação de códigos e gerenciamento de sessões.
 * </p>
 */
public enum SecureTokenType {

  /**
   * Token utilizado para ativação de conta de usuário.
   */
  TOKEN_ATIVACAO(50),

  /**
   * Token utilizado para processos de redefinição de senha.
   */
  TOKEN_REDEFINICAO_SENHA(30),

  /**
   * Código de verificação (geralmente numérico ou alfanumérico curto).
   */
  CODIGO_VERIFICACAO(5),

  /**
   * Token relacionado ao gerenciamento de sessão.
   */
  SESSION(20);

  /**
   * Tamanho do token em bytes.
   */
  private final int byteLength;

  /**
   * Construtor do enum que define o comprimento do token.
   * 
   * @param byteLength o tamanho do token em bytes
   */
  SecureTokenType(int byteLength) {
    this.byteLength = byteLength;
  }

  /**
   * Retorna o comprimento (em bytes) do token associado ao tipo.
   *
   * @return o tamanho do token em bytes
   */
  public int getByteLength() {
    return byteLength;
  }
}
