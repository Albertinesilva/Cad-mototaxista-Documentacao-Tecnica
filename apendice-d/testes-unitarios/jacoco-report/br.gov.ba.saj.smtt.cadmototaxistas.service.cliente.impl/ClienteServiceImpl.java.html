package br.gov.ba.saj.smtt.cadmototaxistas.service.cliente.impl;

import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.security.core.userdetails.User;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Cliente;
import br.gov.ba.saj.smtt.cadmototaxistas.repositories.ClienteRepository;
import br.gov.ba.saj.smtt.cadmototaxistas.security.model.entities.Usuario;
import br.gov.ba.saj.smtt.cadmototaxistas.security.services.usuario.UsuarioService;
import br.gov.ba.saj.smtt.cadmototaxistas.service.cliente.ClienteService;

/**
 * Implementação do serviço para operações relacionadas ao Cliente.
 */
@Service
public class ClienteServiceImpl implements ClienteService {

  @Autowired
  private ClienteRepository clienteRepository;

  @Autowired
  private UsuarioService usuarioService;

  /**
   * Salva um cliente no banco de dados, associando-o ao usuário autenticado
   * após validar a senha fornecida.
   *
   * @param cliente objeto Cliente com os dados a serem salvos.
   * @param user    objeto User com dados do usuário autenticado.
   * @return {@code true} se o cliente foi salvo com sucesso; {@code false} caso a
   *         senha esteja incorreta.
   */
  @Override
  @Transactional
  public boolean salvar(Cliente cliente, User user) {
    Optional<Usuario> usuarioOpt = usuarioService.validarSenhaEObterUsuario(cliente.getUsuario().getSenha(), user);
    if (usuarioOpt.isEmpty()) {
      return false;
    }

    cliente.setUsuario(usuarioOpt.get());
    clienteRepository.save(cliente);
    return true;
  }

  /**
   * Busca um cliente pelo email associado ao usuário.
   *
   * @param email email do usuário associado ao cliente.
   * @return Optional contendo o Cliente encontrado, ou vazio caso não exista.
   */
  @Override
  @Transactional(readOnly = true)
  public Optional<Cliente> buscarPorUsuarioEmail(String email) {
    return clienteRepository.findByUsuarioEmail(email);
  }

  /**
   * Edita os dados de um cliente existente após validar a senha fornecida.
   *
   * @param cliente objeto Cliente com dados atualizados e identificador válido.
   * @param user    objeto User com dados do usuário autenticado.
   * @return {@code true} se a edição foi realizada com sucesso; {@code false}
   *         caso a senha esteja incorreta.
   * @throws IllegalArgumentException se o cliente não for encontrado no banco.
   */
  @Override
  @Transactional
  public boolean editar(Cliente cliente, User user) {
    Optional<Usuario> usuarioOpt = usuarioService.validarSenhaEObterUsuario(cliente.getUsuario().getSenha(), user);
    if (usuarioOpt.isEmpty()) {
      return false;
    }
    Cliente entity = clienteRepository.findById(cliente.getId())
        .orElseThrow(() -> new IllegalArgumentException("Cliente não encontrado"));

    entity.setNome(cliente.getNome());
    entity.setDataNascimento(cliente.getDataNascimento());
    entity.setEndereco(cliente.getEndereco());
    entity.setContato(cliente.getContato());

    clienteRepository.save(entity);
    return true;
  }

}
