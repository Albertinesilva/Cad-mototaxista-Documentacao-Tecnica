package br.gov.ba.saj.smtt.cadmototaxistas.web.controllers.funcionario;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.security.core.annotation.AuthenticationPrincipal;
import org.springframework.security.core.userdetails.User;
import org.springframework.stereotype.Controller;
import org.springframework.ui.ModelMap;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.ModelAttribute;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.servlet.mvc.support.RedirectAttributes;

import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Funcionario;
import br.gov.ba.saj.smtt.cadmototaxistas.model.enums.ETipoSanguineo;
import br.gov.ba.saj.smtt.cadmototaxistas.security.model.entities.Usuario;
import br.gov.ba.saj.smtt.cadmototaxistas.security.services.usuario.UsuarioService;
import br.gov.ba.saj.smtt.cadmototaxistas.security.services.usuario.impl.UsuarioServiceImpl;
import br.gov.ba.saj.smtt.cadmototaxistas.service.funcionario.FuncionarioService;

/**
 * Controller responsável pelo gerenciamento das operações relacionadas aos funcionários.
 *
 * Oferece funcionalidades para visualização, cadastro, edição e atualização dos dados pessoais do funcionário
 * autenticado. Utiliza os serviços de {@link FuncionarioService} e {@link UsuarioServiceImpl} para
 * persistência e recuperação de dados.
 */
@Controller
@RequestMapping("funcionarios")
public class FuncionarioController {

    @Autowired
    private FuncionarioService funcionarioService;

    @Autowired
    private UsuarioService usuarioService;

    /**
     * Exibe a página de dados pessoais do funcionário autenticado.
     * Caso o objeto funcionário não possua ID, busca pelo e-mail do usuário autenticado.
     *
     * @param funcionario objeto funcionário para edição
     * @param model objeto de modelo para passagem de atributos à view
     * @param user usuário autenticado via Spring Security
     * @return nome da view para cadastro/edição do funcionário
     */
    @GetMapping({ "/dados" })
    public String abrirPorFuncionario(Funcionario funcionario, ModelMap model, @AuthenticationPrincipal User user) {
        if (funcionario.hasNotId()) {
            funcionario = funcionarioService.buscarPorEmail(user.getUsername());
            model.addAttribute("funcionario", funcionario);
        }
        return "funcionario/cadastro";
    }

    /**
     * Salva um novo funcionário ou atualiza um existente.
     * Se o funcionário e seu usuário ainda não possuírem ID, associa o usuário autenticado.
     *
     * @param funcionario objeto funcionário a ser salvo
     * @param attr atributos para mensagens flash na redirect
     * @param user usuário autenticado via Spring Security
     * @return redirecionamento para a página de dados do funcionário
     */
    @PostMapping("/salvar")
    public String salvar(Funcionario funcionario, RedirectAttributes attr, @AuthenticationPrincipal User user) {
        if (funcionario.hasNotId() && funcionario.getUsuario().hasNotId()) {
            Usuario usuario = usuarioService.buscarPorEmail(user.getUsername());
            funcionario.setUsuario(usuario);
        }
        funcionarioService.salvar(funcionario);
        attr.addFlashAttribute("success", "Funcionário inserido com sucesso.");
        attr.addFlashAttribute("funcionario", funcionario);
        return "redirect:/funcionarios/dados";
    }

    /**
     * Edita os dados de um funcionário existente.
     *
     * @param funcionario objeto funcionário a ser editado
     * @param attr atributos para mensagens flash na redirect
     * @return redirecionamento para a página de dados do funcionário
     */
    @PostMapping({ "/editar" })
    public String editar(Funcionario funcionario, RedirectAttributes attr) {
        funcionarioService.editar(funcionario);
        attr.addFlashAttribute("success", "Funcionário editado com sucesso.");
        attr.addFlashAttribute("funcionario", funcionario);
        return "redirect:/funcionarios/dados";
    }

    /**
     * Fornece os tipos sanguíneos disponíveis para uso em formulários.
     *
     * @return array contendo os valores do enum {@link ETipoSanguineo}
     */
    @ModelAttribute("tipoSanguineos")
    public ETipoSanguineo[] getETipoSanguineos() {
        return ETipoSanguineo.values();
    }

}
