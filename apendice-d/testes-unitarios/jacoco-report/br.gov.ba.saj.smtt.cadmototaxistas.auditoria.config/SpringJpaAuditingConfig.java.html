package br.gov.ba.saj.smtt.cadmototaxistas.auditoria.config;

import org.springframework.context.annotation.Configuration;
import org.springframework.data.domain.AuditorAware;
import org.springframework.data.jpa.repository.config.EnableJpaAuditing;
import org.springframework.lang.NonNull;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.context.SecurityContextHolder;

import java.util.Optional;

/**
 * Configuração para habilitar a auditoria automática no JPA.
 * 
 * Essa classe implementa {@link AuditorAware} para informar o usuário atual
 * autenticado
 * que será usado como auditor nas entidades auditadas (ex: criação,
 * atualização).
 * 
 * Habilita o recurso de auditoria via {@link EnableJpaAuditing}.
 * 
 * @author SeuNome
 * @since 1.0
 */
@EnableJpaAuditing
@Configuration
public class SpringJpaAuditingConfig implements AuditorAware<String> {

  /**
   * Retorna o nome do usuário atualmente autenticado para ser usado
   * como auditor em operações JPA auditadas.
   * 
   * Caso não haja autenticação válida, retorna {@link Optional#empty()}.
   * 
   * @return um {@link Optional} contendo o nome do usuário autenticado ou vazio se não autenticado.
   */
  @Override
  @NonNull
  public Optional<String> getCurrentAuditor() {
    Authentication authentication = SecurityContextHolder.getContext().getAuthentication();

    if (authentication != null && authentication.isAuthenticated()) {
      return Optional.of(authentication.getName());
    }
    return Optional.empty();
  }
}
