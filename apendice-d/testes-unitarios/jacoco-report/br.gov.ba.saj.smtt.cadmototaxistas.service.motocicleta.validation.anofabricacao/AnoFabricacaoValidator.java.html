package br.gov.ba.saj.smtt.cadmototaxistas.service.motocicleta.validation.anofabricacao;

import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;

import java.time.LocalDate;

/**
 * Validador para verificar se o ano de fabricação de uma motocicleta
 * é válido conforme regra de negócio: não pode ser mais antigo que 10 anos da data atual.
 */
public class AnoFabricacaoValidator implements ConstraintValidator<ValidAnoFabricacao, LocalDate> {

    /**
     * Valida se o ano de fabricação está dentro do limite mínimo permitido (últimos 10 anos).
     * 
     * @param anoFabricacao data do ano de fabricação a ser validada
     * @param context contexto da validação
     * @return true se o anoFabricacao for nulo (delegado a @NotNull) ou estiver dentro do limite; false caso contrário
     */
    @Override
    public boolean isValid(LocalDate anoFabricacao, ConstraintValidatorContext context) {
        if (anoFabricacao == null) {
            return true; // Deixa @NotNull cuidar da nulidade
        }

        LocalDate hoje = LocalDate.now();
        LocalDate limiteMinimo = hoje.minusYears(10);

        return anoFabricacao.isAfter(limiteMinimo) || anoFabricacao.isEqual(limiteMinimo);
    }
}
