package br.gov.ba.saj.smtt.cadmototaxistas.model.enums.converter;

import br.gov.ba.saj.smtt.cadmototaxistas.model.enums.ETipoSanguineo;
import jakarta.persistence.AttributeConverter;
import jakarta.persistence.Converter;

/**
 * Conversor JPA responsável por mapear a enumeração {@link ETipoSanguineo}
 * para sua representação textual no banco de dados e vice-versa.
 * 
 * <p>
 * Essa classe é anotada com {@code @Converter(autoApply = true)}, o que
 * significa que ela será aplicada automaticamente em todos os atributos do tipo
 * {@link ETipoSanguineo}.
 * </p>
 *
 * <p>Exemplo de persistência:</p>
 * <ul>
 *   <li>{@code ETipoSanguineo.APOSITIVO} → "A+"</li>
 * </ul>
 *
 * <p>Exemplo de leitura:</p>
 * <ul>
 *   <li>"O-" → {@code ETipoSanguineo.ONEGATIVO}</li>
 * </ul>
 *
 * @author Albert Silva
 */
@Converter(autoApply = true)
public class TipoSanguineoConverter implements AttributeConverter<ETipoSanguineo, String> {

  /**
   * Converte o atributo {@link ETipoSanguineo} da entidade para seu valor em
   * string,
   * que será armazenado no banco de dados.
   *
   * @param attribute a enumeração {@link ETipoSanguineo}, pode ser {@code null}
   * @return uma string representando o tipo sanguíneo (ex: "A+"), ou {@code null}
   *         se o atributo for nulo
   */
  @Override
  public String convertToDatabaseColumn(ETipoSanguineo attribute) {
    return attribute != null ? attribute.getTipoSanguineo() : null;
  }

  /**
   * Converte o valor textual vindo do banco de dados para a enumeração
   * {@link ETipoSanguineo}.
   *
   * @param dbData a string armazenada no banco de dados, pode ser {@code null}
   * @return a enumeração correspondente, ou {@code null} se o dado do banco for
   *         nulo
   * @throws IllegalArgumentException se o valor não corresponder a nenhum tipo
   *                                  válido
   */
  @Override
  public ETipoSanguineo convertToEntityAttribute(String dbData) {
    return dbData != null ? ETipoSanguineo.fromValue(dbData) : null;
  }
}
