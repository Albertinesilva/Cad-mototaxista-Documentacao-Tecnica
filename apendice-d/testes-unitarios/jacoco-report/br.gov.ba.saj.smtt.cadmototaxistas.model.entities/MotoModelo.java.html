package br.gov.ba.saj.smtt.cadmototaxistas.model.entities;

import java.util.ArrayList;
import java.util.List;

import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.JoinColumn;
import jakarta.persistence.ManyToOne;
import jakarta.persistence.OneToMany;
import jakarta.persistence.Table;
import jakarta.validation.Valid;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Size;

import lombok.Getter;
import lombok.Setter;

/**
 * Representa o modelo de uma motocicleta.
 * <p>
 * Contém informações detalhadas do modelo, incluindo nome, descrição,
 * status ativo, a marca associada e a lista de motocicletas deste modelo.
 * </p>
 * 
 * Estabelece relacionamento ManyToOne com {@link MotoMarca} e
 * OneToMany com {@link Motocicleta}.
 * 
 * @author
 */
@Getter
@Setter
@Entity
@Table(name = "moto_modelo")
public class MotoModelo extends AbstractEntity {

    /**
     * Nome do modelo da motocicleta.
     * <p>
     * Obrigatório, único, com tamanho entre 2 e 60 caracteres.
     * </p>
     */
    @NotBlank(message = "{NotBlank.motoModelo.nome}")
    @Size(min = 2, max = 60, message = "{Size.motoModelo.nome}")
    @Column(name = "modelo_moto", nullable = false, unique = true, length = 100)
    private String nomeModelo;

    /**
     * Descrição opcional do modelo da motocicleta.
     */
    @Column(name = "descricao_modelo", nullable = true)
    private String descricaoModelo;

    /**
     * Indica se o modelo está ativo.
     */
    @Column(name = "ativo", nullable = false)
    private boolean ativo;

    /**
     * Marca associada a este modelo de motocicleta.
     * <p>
     * Obrigatório e validado.
     * </p>
     */
    @Valid
    @NotNull(message = "{NotNull.motoModelo.marca}")
    @ManyToOne
    @JoinColumn(name = "marca_fk")
    private MotoMarca marcaMoto;

    /**
     * Lista de motocicletas que utilizam este modelo.
     * <p>
     * Relacionamento OneToMany mapeado pelo atributo 'motoModelo' na entidade
     * {@link Motocicleta}.
     * </p>
     */
    @OneToMany(mappedBy = "motoModelo")
    private List<Motocicleta> motocicleta = new ArrayList<>();

}
