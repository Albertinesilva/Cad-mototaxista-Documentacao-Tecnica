package br.gov.ba.saj.smtt.cadmototaxistas.model.entities;

import java.time.LocalDate;

import org.springframework.format.annotation.DateTimeFormat;
import org.springframework.format.annotation.DateTimeFormat.ISO;

import jakarta.persistence.CascadeType;
import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.EnumType;
import jakarta.persistence.Enumerated;
import jakarta.persistence.OneToOne;
import jakarta.persistence.Table;
import jakarta.validation.constraints.Future;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Size;

import br.gov.ba.saj.smtt.cadmototaxistas.model.enums.ECategoria;

import lombok.Getter;
import lombok.Setter;

/**
 * Representa a Carteira Nacional de Habilitação (CNH) de um condutor.
 * <p>
 * Esta entidade contém informações como número da habilitação, categoria da
 * CNH,
 * data de vencimento, e o condutor associado.
 * </p>
 * 
 * As validações garantem que os campos estejam preenchidos corretamente e que
 * a data de vencimento seja futura.
 * 
 * @author
 */
@Getter
@Setter
@Entity
@Table(name = "cnh")
public class Cnh extends AbstractEntity {

    /**
     * Número da Carteira Nacional de Habilitação.
     * <p>
     * Deve ter exatamente 11 caracteres, ser único e não pode ser nulo ou vazio.
     * </p>
     */
    @NotBlank(message = "{NotBlank.cnh.numero}")
    @Size(min = 11, max = 11, message = "{Size.cnh.numero}")
    @Column(name = "numero_habilitacao", nullable = false, unique = true, length = 11)
    private String numero;

    /**
     * Categoria da CNH, representada pelo enum {@link ECategoria}.
     * <p>
     * Campo obrigatório.
     * </p>
     */
    @NotNull(message = "{NotNull.cnh.categoria}")
    @Enumerated(EnumType.STRING)
    @Column(name = "categoria", nullable = false, length = 2)
    private ECategoria categoria;

    /**
     * Data de vencimento da CNH.
     * <p>
     * Deve ser uma data futura e obrigatória.
     * </p>
     */
    @NotNull(message = "{NotNull.cnh.dataVencimento}")
    @Future(message = "{Future.cnh.dataVencimento}")
    @Column(name = "data_vencimento", nullable = false, columnDefinition = "DATE")
    @DateTimeFormat(iso = ISO.DATE)
    private LocalDate dataVencimento;

    /**
     * Relação bidirecional com o condutor associado à CNH.
     * <p>
     * Mapeado pelo atributo {@code cnh} na entidade {@link Condutor}.
     * O cascade {@link CascadeType#ALL} propaga todas as operações.
     * </p>
     */
    @OneToOne(mappedBy = "cnh", cascade = CascadeType.ALL)
    private Condutor condutor;

}
