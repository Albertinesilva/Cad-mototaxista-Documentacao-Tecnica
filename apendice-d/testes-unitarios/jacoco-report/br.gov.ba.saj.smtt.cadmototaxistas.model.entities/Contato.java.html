package br.gov.ba.saj.smtt.cadmototaxistas.model.entities;

import br.gov.ba.saj.smtt.cadmototaxistas.service.condutor.validation.contato.ValidTelefone;
import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.Table;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.Size;
import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * Entidade que representa os dados de contato de um condutor.
 * <p>
 * Contém números de telefone principal obrigatório e telefone comercial
 * opcional.
 * O telefone comercial é validado por uma validação customizada
 * {@link ValidTelefone}.
 * </p>
 * <p>
 * Possui método utilitário para retornar o telefone comercial apenas com os
 * dígitos numéricos,
 * removendo quaisquer caracteres especiais.
 * </p>
 * 
 * @author
 */
@AllArgsConstructor
@NoArgsConstructor
@Getter
@Setter
@Entity
@Table(name = "contato")
public class Contato extends AbstractEntity {

    /**
     * Telefone principal obrigatório do condutor.
     * Deve conter exatamente 15 caracteres e ser único.
     */
    @NotBlank(message = "{NotBlank.contato.telefonePrincipal}")
    @Size(min = 15, max = 15, message = "{Size.contato.telefonePrincipal}")
    @Column(name = "telefone_principal", nullable = false, unique = true, length = 15)
    private String telefonePrincipal;

    /**
     * Telefone comercial opcional do condutor.
     * Validado com anotação customizada {@link ValidTelefone}.
     * Pode ser nulo.
     */
    @ValidTelefone
    @Column(name = "telefone_comercial", nullable = true, unique = true, length = 15)
    private String telefoneComercial = null;

    /**
     * Retorna o número do telefone principal contendo apenas os dígitos numéricos.
     * Remove todos os caracteres especiais como parênteses, traços, espaços, etc.
     *
     * @return String contendo apenas os números do telefone principal.
     *         Retorna string vazia se o telefone principal for {@code null}.
     */
    public String getTelefonePrincipalLimpo() {
        if (telefonePrincipal == null)
            return "";
        return telefonePrincipal.replaceAll("[^\\d]", "");
    }

    /**
     * Retorna o número do telefone comercial contendo apenas os dígitos numéricos.
     * Remove todos os caracteres especiais como parênteses, traços, espaços, etc.
     * 
     * @return String contendo apenas os números do telefone comercial.
     *         Retorna string vazia se o telefone comercial for {@code null}.
     */
    public String getTelefoneComercialLimpo() {
        if (telefoneComercial == null)
            return "";
        return telefoneComercial.replaceAll("[^\\d]", "");
    }

    // Setter que normaliza valores vazios para null
    public void setTelefoneComercial(String telefoneComercial) {
        if (telefoneComercial != null && telefoneComercial.trim().isEmpty()) {
            this.telefoneComercial = null;
        } else {
            this.telefoneComercial = telefoneComercial;
        }
    }

}
