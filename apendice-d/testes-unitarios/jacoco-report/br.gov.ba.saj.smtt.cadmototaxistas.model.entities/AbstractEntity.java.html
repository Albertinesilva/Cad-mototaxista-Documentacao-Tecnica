package br.gov.ba.saj.smtt.cadmototaxistas.model.entities;

import java.io.Serializable;

import jakarta.persistence.GeneratedValue;
import jakarta.persistence.GenerationType;
import jakarta.persistence.Id;
import jakarta.persistence.MappedSuperclass;

/**
 * Classe base abstrata para todas as entidades do sistema que possuem um
 * identificador único.
 * <p>
 * Esta classe fornece a implementação padrão do atributo {@code id} e dos
 * métodos
 * {@code equals}, {@code hashCode} e {@code toString} baseados nesse
 * identificador.
 * <p>
 * Deve ser estendida por todas as entidades JPA que precisam de um
 * identificador
 * do tipo {@link Long} gerado automaticamente.
 * </p>
 * 
 * @author
 */
@MappedSuperclass
public abstract class AbstractEntity implements Serializable {

	private static final long serialVersionUID = 1L;

	/**
	 * Identificador único da entidade.
	 * <p>
	 * É gerado automaticamente pelo provedor JPA usando a estratégia
	 * {@link GenerationType#IDENTITY}.
	 * </p>
	 */
	@Id
	@GeneratedValue(strategy = GenerationType.IDENTITY)
	private Long id;

	/**
	 * Construtor padrão.
	 */
	public AbstractEntity() {
		super();
	}

	/**
	 * Retorna o identificador único da entidade.
	 * 
	 * @return id da entidade
	 */
	public Long getId() {
		return id;
	}

	/**
	 * Define o identificador único da entidade.
	 * 
	 * @param id o identificador a ser setado
	 */
	public void setId(Long id) {
		this.id = id;
	}

	/**
	 * Verifica se a entidade ainda não possui um {@code id} atribuído.
	 * 
	 * @return {@code true} se o {@code id} for {@code null}, indicando que a
	 *         entidade
	 *         é nova e não foi persistida ainda; {@code false} caso contrário.
	 */
	public boolean hasNotId() {
		return id == null;
	}

	/**
	 * Verifica se a entidade possui um {@code id} atribuído.
	 * 
	 * @return {@code true} se o {@code id} não for {@code null}, indicando que a
	 *         entidade foi persistida; {@code false} caso contrário.
	 */
	public boolean hasId() {
		return id != null;
	}

	/**
	 * Gera o código hash da entidade baseado no seu {@code id}.
	 * 
	 * @return código hash da entidade
	 */
	@Override
	public int hashCode() {
		final int prime = 31;
		int result = 1;
		result = prime * result + ((id == null) ? 0 : id.hashCode());
		return result;
	}

	/**
	 * Compara esta entidade com outro objeto para verificar igualdade.
	 * <p>
	 * Duas entidades são consideradas iguais se forem da mesma classe e
	 * possuírem o mesmo {@code id}.
	 * </p>
	 * 
	 * @param obj objeto a ser comparado
	 * @return {@code true} se forem iguais; {@code false} caso contrário
	 */
	@Override
	public boolean equals(Object obj) {
		if (this == obj)
			return true;
		if (obj == null)
			return false;
		if (getClass() != obj.getClass())
			return false;
		AbstractEntity other = (AbstractEntity) obj;
		if (id == null) {
			if (other.id != null)
				return false;
		} else if (!id.equals(other.id))
			return false;
		return true;
	}

	/**
	 * Retorna uma representação em texto da entidade com o nome da classe e o
	 * identificador.
	 * 
	 * @return string com informações da entidade
	 */
	@Override
	public String toString() {
		return String.format("Entidade %s id: %s", this.getClass().getName(), getId());
	}

}
