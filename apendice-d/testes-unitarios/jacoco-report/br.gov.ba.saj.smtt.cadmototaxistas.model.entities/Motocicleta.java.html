package br.gov.ba.saj.smtt.cadmototaxistas.model.entities;

import java.time.LocalDate;

import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.EnumType;
import jakarta.persistence.Enumerated;
import jakarta.persistence.JoinColumn;
import jakarta.persistence.ManyToOne;
import jakarta.persistence.OneToOne;
import jakarta.persistence.Table;
import jakarta.validation.Valid;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.PastOrPresent;
import jakarta.validation.constraints.Size;

import org.springframework.format.annotation.DateTimeFormat;
import org.springframework.format.annotation.DateTimeFormat.ISO;

import br.gov.ba.saj.smtt.cadmototaxistas.model.enums.ESituacaoMoto;
import br.gov.ba.saj.smtt.cadmototaxistas.model.enums.ETipoPlaca;
import br.gov.ba.saj.smtt.cadmototaxistas.service.motocicleta.validation.anofabricacao.ValidAnoFabricacao;
import br.gov.ba.saj.smtt.cadmototaxistas.service.motocicleta.validation.anomodelo.ValidAnoModelo;
import lombok.Getter;
import lombok.Setter;

/**
 * Entidade que representa uma motocicleta associada a um condutor.
 * <p>
 * Contém informações como placa, tipo de placa, situação da moto, ano de
 * fabricação, ano do modelo,
 * observações, modelo da moto e o condutor relacionado.
 * </p>
 * <p>
 * As validações customizadas {@link ValidAnoFabricacao} e
 * {@link ValidAnoModelo} garantem a consistência das datas.
 * </p>
 * 
 * @author
 */
@ValidAnoModelo
@Getter
@Setter
@Entity
@Table(name = "motocicleta")
public class Motocicleta extends AbstractEntity {

    /**
     * Placa da motocicleta.
     * Obrigatória, única, mínimo 7 caracteres.
     */
    @NotBlank(message = "{NotBlank.motocicleta.placa}")
    @Size(min = 7, message = "{Size.motocicleta.placa}")
    @Column(name = "placa", nullable = false, unique = true, length = 10)
    private String placa;

    /**
     * Tipo da placa (ex: Mercosul, padrão).
     * Opcional, armazenado como String do enum {@link ETipoPlaca}.
     */
    @Column(name = "tipo_placa", nullable = true, length = 8)
    @Enumerated(EnumType.STRING)
    private ETipoPlaca tipoPlaca;

    /**
     * Situação da motocicleta.
     * Opcional, armazenado como String do enum {@link ESituacaoMoto}.
     */
    @Column(name = "situacao_moto", nullable = true, length = 13)
    @Enumerated(EnumType.STRING)
    private ESituacaoMoto ePropria;

    /**
     * Ano de fabricação da motocicleta.
     * Obrigatório, deve ser data passada ou presente.
     * Validado pela anotação customizada {@link ValidAnoFabricacao}.
     */
    @NotNull(message = "{NotNull.motocicleta.anoFabricacao}")
    @PastOrPresent(message = "{PastOrPresent.motocicleta.anoFabricacao}")
    @ValidAnoFabricacao
    @DateTimeFormat(iso = ISO.DATE)
    @Column(name = "ano_fabricacao", nullable = false, columnDefinition = "DATE")
    private LocalDate anoFabricacao;

    /**
     * Ano do modelo da motocicleta.
     * Obrigatório.
     * Validado pela anotação customizada {@link ValidAnoModelo}.
     */
    @NotNull(message = "{NotNull.motocicleta.anoModelo}")
    @DateTimeFormat(iso = ISO.DATE)
    @Column(name = "ano_modelo", nullable = false, columnDefinition = "DATE")
    private LocalDate anoModelo;

    /**
     * Observações adicionais sobre a motocicleta.
     * Opcional, até 500 caracteres.
     */
    @Column(name = "observacao", nullable = true, length = 500)
    private String observacao;

    /**
     * Modelo da motocicleta.
     * Obrigatório e validado.
     */
    @Valid
    @NotNull(message = "{NotNull.motocicleta.motoModelo}")
    @ManyToOne
    @JoinColumn(name = "moto_modelo_fk")
    private MotoModelo motoModelo;

    /**
     * Condutor associado a esta motocicleta.
     * Relacionamento bidirecional OneToOne mapeado pela propriedade 'motocicleta'
     * do condutor.
     */
    @OneToOne(mappedBy = "motocicleta")
    private Condutor condutor;

}
