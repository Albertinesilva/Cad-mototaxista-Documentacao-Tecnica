package br.gov.ba.saj.smtt.cadmototaxistas.model.entities;

import java.util.ArrayList;
import java.util.List;

import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.OneToMany;
import jakarta.persistence.Table;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.Size;

import lombok.Getter;
import lombok.Setter;

/**
 * Representa a marca de motocicletas.
 * <p>
 * Contém informações básicas sobre a marca, como nome, descrição, status ativo,
 * e a lista de modelos de motocicletas associados.
 * </p>
 * 
 * A relação com {@link MotoModelo} é unidirecional, com mapeamento OneToMany.
 * 
 * @author
 */
@Getter
@Setter
@Entity
@Table(name = "moto_marca")
public class MotoMarca extends AbstractEntity {

    /**
     * Nome da marca da motocicleta.
     * <p>
     * Obrigatório, único, com tamanho entre 3 e 20 caracteres.
     * </p>
     */
    @NotBlank(message = "{NotBlank.motoMarca.nome}")
    @Size(min = 3, max = 20, message = "{Size.motoMarca.nome}")
    @Column(name = "nome_marca", nullable = false, unique = true, length = 20)
    private String nomeMarca;

    /**
     * Descrição opcional da marca da motocicleta.
     */
    @Column(name = "descricao_marca", nullable = true, length = 255)
    private String descricaoMarca;

    /**
     * Indica se a marca está ativa.
     */
    @Column(name = "ativo", nullable = false)
    private boolean ativo;

    /**
     * Lista dos modelos de motocicletas associados a esta marca.
     * <p>
     * Relacionamento OneToMany mapeado pelo atributo 'marcaMoto' da entidade
     * {@link MotoModelo}.
     * </p>
     */
    @OneToMany(mappedBy = "marcaMoto")
    private List<MotoModelo> motoModelo = new ArrayList<>();

}
