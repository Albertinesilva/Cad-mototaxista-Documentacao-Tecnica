package br.gov.ba.saj.smtt.cadmototaxistas.model.entities;

import java.time.LocalDate;

import jakarta.persistence.CascadeType;
import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.EnumType;
import jakarta.persistence.Enumerated;
import jakarta.persistence.JoinColumn;
import jakarta.persistence.OneToOne;
import jakarta.persistence.Table;

import org.hibernate.validator.constraints.br.CPF;
import org.springframework.format.annotation.DateTimeFormat;
import org.springframework.format.annotation.DateTimeFormat.ISO;

import br.gov.ba.saj.smtt.cadmototaxistas.model.enums.ETipoSanguineo;
import br.gov.ba.saj.smtt.cadmototaxistas.security.model.entities.Usuario;
import lombok.Getter;
import lombok.Setter;

/**
 * Representa um funcionário do sistema.
 * <p>
 * Contém informações pessoais e dados de cadastro, incluindo nome, CPF,
 * data de nascimento, matrícula, tipo sanguíneo e vínculo com o usuário do
 * sistema.
 * </p>
 * 
 * Possui relacionamento OneToOne com a entidade {@link Usuario}.
 * 
 * @author
 */
@Getter
@Setter
@Entity
@Table(name = "funcionario")
public class Funcionario extends AbstractEntity {

	/**
	 * Nome completo do funcionário.
	 */
	@Column(name = "nome", nullable = false, length = 255)
	private String nome;

	/**
	 * CPF do funcionário.
	 * <p>
	 * Valor único e válido conforme a validação {@link CPF}.
	 * </p>
	 */
	@CPF(message = "{CPF.ServidorSmtt.cpf}")
	@Column(name = "cpf", nullable = true, unique = true, length = 15)
	private String cpf;

	/**
	 * Data de nascimento do funcionário.
	 */
	@DateTimeFormat(iso = ISO.DATE)
	@Column(name = "data_nascimento", nullable = true, columnDefinition = "DATE")
	private LocalDate dataNascimento;

	/**
	 * Data de inscrição do funcionário no sistema.
	 */
	@DateTimeFormat(iso = ISO.DATE)
	@Column(name = "data_inscricao", nullable = true, columnDefinition = "DATE")
	private LocalDate dtInscricao;

	/**
	 * Matrícula funcional do funcionário.
	 * <p>
	 * Valor único.
	 * </p>
	 */
	@Column(name = "matricula", nullable = true, unique = true, length = 11)
	private String matricula;

	/**
	 * Tipo sanguíneo do funcionário.
	 */
	@Enumerated(EnumType.STRING)
	@Column(name = "tipo_sanguineo", nullable = true, length = 15)
	private ETipoSanguineo tipoSanguineo;

	/**
	 * Usuário do sistema vinculado a este funcionário.
	 * <p>
	 * Relacionamento OneToOne com cascade REMOVE para exclusão em cascata.
	 * </p>
	 */
	@OneToOne(cascade = CascadeType.REMOVE)
	@JoinColumn(name = "id_usuario")
	private Usuario usuario;

	/**
	 * Construtor padrão.
	 */
	public Funcionario() {
		super();
	}

	/**
	 * Construtor que seta o ID da entidade.
	 * 
	 * @param id Identificador do funcionário.
	 */
	public Funcionario(Long id) {
		super.setId(id);
	}

	/**
	 * Construtor que seta o usuário vinculado.
	 * 
	 * @param usuario Usuário do sistema associado.
	 */
	public Funcionario(Usuario usuario) {
		this.usuario = usuario;
	}

}
