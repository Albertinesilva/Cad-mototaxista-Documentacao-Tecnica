package br.gov.ba.saj.smtt.cadmototaxistas.model.entities;

import java.time.LocalDate;
import java.time.LocalDateTime;

import org.hibernate.validator.constraints.br.CPF;
import org.springframework.data.annotation.CreatedBy;
import org.springframework.data.annotation.CreatedDate;
import org.springframework.data.annotation.LastModifiedBy;
import org.springframework.data.annotation.LastModifiedDate;
import org.springframework.data.jpa.domain.support.AuditingEntityListener;
import org.springframework.format.annotation.DateTimeFormat;
import org.springframework.format.annotation.DateTimeFormat.ISO;

import br.gov.ba.saj.smtt.cadmototaxistas.model.enums.EModalidadeDeCondutor;
import br.gov.ba.saj.smtt.cadmototaxistas.model.enums.ETipoSanguineo;
import br.gov.ba.saj.smtt.cadmototaxistas.model.enums.converter.TipoSanguineoConverter;
import br.gov.ba.saj.smtt.cadmototaxistas.model.interfaces.EntidadeComContato;
import br.gov.ba.saj.smtt.cadmototaxistas.service.condutor.validation.idademinima.ValidIdadeMinima;
import jakarta.persistence.CascadeType;
import jakarta.persistence.Column;
import jakarta.persistence.Convert;
import jakarta.persistence.Entity;
import jakarta.persistence.EntityListeners;
import jakarta.persistence.EnumType;
import jakarta.persistence.Enumerated;
import jakarta.persistence.JoinColumn;
import jakarta.persistence.ManyToOne;
import jakarta.persistence.OneToOne;
import jakarta.persistence.Table;
import jakarta.validation.Valid;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Size;

import lombok.Getter;
import lombok.Setter;

/**
 * Entidade que representa um Condutor dentro do sistema.
 * <p>
 * Contém informações pessoais, documentos, contatos, endereço, veículos
 * associados,
 * além de dados de auditoria automática como criação e modificação.
 * </p>
 * <p>
 * Possui validações para garantir a integridade dos dados, como CPF válido,
 * idade mínima e campos obrigatórios.
 * </p>
 * 
 * @author
 */
@Getter
@Setter
@Entity
@Table(name = "condutor")
@EntityListeners(AuditingEntityListener.class)
public class Condutor extends AbstractEntity implements EntidadeComContato {

    /**
     * Número de ordem único que identifica o condutor.
     * Deve ter entre 1 e 10 caracteres.
     */
    @Column(name = "numero_ordem", nullable = true, unique = true, length = 10)
    private String numOrdem;

    /**
     * Nome completo do condutor.
     * Deve ter no mínimo 2 caracteres e não pode ser nulo ou vazio.
     */
    @NotBlank(message = "{NotBlank.condutor.nome}")
    @Size(min = 2, message = "{Size.condutor.nome}")
    @Column(name = "nome", nullable = false, length = 255)
    private String nome;

    /**
     * Modalidade do condutor, representada pelo enum {@link EModalidadeDeCondutor}.
     */
    @Enumerated(EnumType.STRING)
    @Column(name = "modalidade_condutor", nullable = false, length = 12)
    private EModalidadeDeCondutor modalidadeDeCondutor;

    /**
     * Apelido do condutor, campo opcional.
     */
    @Column(name = "apelido", nullable = true, length = 50)
    private String apelido;

    /**
     * Registro Geral (RG) do condutor.
     * Deve ter entre 5 e 15 caracteres e ser único.
     */
    @NotBlank(message = "{NotBlank.condutor.rg}")
    @Size(min = 5, max = 15, message = "{Size.condutor.rg}")
    @Column(name = "rg", nullable = true, unique = true, length = 15)
    private String rg;

    /**
     * Cadastro de Pessoa Física (CPF) do condutor.
     * Campo obrigatório, único e válido conforme validação {@link CPF}.
     */
    @CPF(message = "{CPF.condutor.cpf}")
    @NotBlank(message = "{NotBlank.condutor.cpf}")
    @Size(min = 11, message = "{Size.condutor.cpf}")
    @Column(name = "cpf", nullable = false, unique = true, length = 15)
    private String cpf;

    /**
     * Data de nascimento do condutor.
     * Deve respeitar a idade mínima de 21 anos conforme validação personalizada
     * {@link ValidIdadeMinima}.
     */
    @ValidIdadeMinima(value = 21, message = "{IdadeMinima.condutor.dataNascimento}")
    @NotNull(message = "{NotNull.condutor.dataNascimento}")
    @DateTimeFormat(iso = ISO.DATE)
    @Column(name = "data_nascimento", nullable = false, columnDefinition = "DATE")
    private LocalDate dataNascimento;

    /**
     * Nome do pai do condutor (opcional).
     */
    @Column(name = "nome_pai", nullable = true, length = 255)
    private String paiNome;

    /**
     * Nome da mãe do condutor (opcional).
     */
    @Column(name = "nome_mae", nullable = true, length = 255)
    private String maeNome;

    /**
     * Indica se o cadastro está ativo.
     */
    @Column(name = "cadastro_ativo", nullable = false)
    private boolean cadastroAtivo;

    /**
     * Tipo sanguíneo do condutor.
     * Utiliza conversor para armazenar corretamente no banco.
     */
    @Convert(converter = TipoSanguineoConverter.class)
    @Column(name = "tipo_sanguineo", nullable = true, length = 11)
    private ETipoSanguineo tipoSanguineo;

    /**
     * Data de vencimento do alvará do condutor (opcional).
     */
    @DateTimeFormat(iso = ISO.DATE)
    @Column(name = "data_vencimento_alvara", nullable = true, columnDefinition = "DATE")
    private LocalDate dataVencimentoAlvara;

    /**
     * Data de criação do registro, preenchida automaticamente pelo Spring Data JPA.
     */
    @CreatedDate
    @Column(name = "data_criacao")
    private LocalDateTime dataCriacao;

    /**
     * Data da última modificação do registro, preenchida automaticamente.
     */
    @LastModifiedDate
    @Column(name = "data_modificacao")
    private LocalDateTime dataModificacao;

    /**
     * Usuário que criou o registro, preenchido automaticamente pelo Spring
     * Security.
     */
    @CreatedBy
    @Column(name = "criado_por")
    private String criadorPor;

    /**
     * Usuário que modificou o registro pela última vez.
     */
    @LastModifiedBy
    @Column(name = "modificado_por")
    private String modificadoPor;

    /**
     * Endereço do condutor.
     * Mapeado em relação OneToOne, com cascata para operações.
     */
    @Valid
    @OneToOne(cascade = CascadeType.ALL)
    @JoinColumn(name = "endereco_fk")
    private Endereco endereco;

    /**
     * Ponto (local de trabalho ou referência) associado ao condutor.
     * Relacionamento ManyToOne.
     */
    @ManyToOne
    @JoinColumn(name = "ponto_fk")
    private Ponto ponto;

    /**
     * Contato do condutor, mapeado OneToOne com cascata.
     */
    @Valid
    @OneToOne(cascade = CascadeType.ALL)
    @JoinColumn(name = "contato_fk")
    private Contato contato;

    /**
     * Motocicleta associada ao condutor.
     * Mapeamento OneToOne com cascata.
     */
    @Valid
    @OneToOne(cascade = CascadeType.ALL)
    @JoinColumn(name = "motocicleta_fk")
    private Motocicleta motocicleta;

    /**
     * CNH (Carteira Nacional de Habilitação) do condutor.
     * Mapeamento OneToOne com cascata.
     */
    @Valid
    @OneToOne(cascade = CascadeType.ALL)
    @JoinColumn(name = "cnh_fk")
    private Cnh cnh;

}
