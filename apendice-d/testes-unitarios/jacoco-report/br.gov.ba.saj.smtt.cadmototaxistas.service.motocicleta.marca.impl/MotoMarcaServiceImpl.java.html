package br.gov.ba.saj.smtt.cadmototaxistas.service.motocicleta.marca.impl;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.MotoMarca;
import br.gov.ba.saj.smtt.cadmototaxistas.repositories.MotoMarcaRepository;
import br.gov.ba.saj.smtt.cadmototaxistas.service.motocicleta.marca.MotoMarcaService;
import br.gov.ba.saj.smtt.cadmototaxistas.util.paginacao.PaginacaoUtil;
import br.gov.ba.saj.smtt.cadmototaxistas.util.string.StringUtil;
import jakarta.persistence.EntityManager;
import jakarta.persistence.PersistenceContext;

/**
 * Implementação da interface {@link MotoMarcaService}, responsável por fornecer
 * os serviços relacionados às entidades {@link MotoMarca}.
 *
 * Esta classe é anotada com {@code @Service} e {@code @Transactional} para
 * permitir
 * a gestão transacional e o controle do ciclo de vida dos métodos de serviço.
 */
@Service
@Transactional(readOnly = false)
public class MotoMarcaServiceImpl implements MotoMarcaService {

    @Autowired
    private MotoMarcaRepository motoMarcaRepository;

    @PersistenceContext
    private EntityManager entityManager;

    /**
     * Salva uma nova marca de motocicleta no banco de dados.
     *
     * @param motoMarca objeto {@link MotoMarca} a ser persistido
     */
    @Override
    public void salvar(MotoMarca motoMarca) {
        motoMarcaRepository.save(motoMarca);
    }

    /**
     * Atualiza os dados de uma marca de motocicleta já existente.
     *
     * @param motoMarca objeto {@link MotoMarca} com os dados atualizados
     */
    @Override
    public void editar(MotoMarca motoMarca) {
        motoMarcaRepository.save(motoMarca);
    }

    /**
     * Remove uma marca de motocicleta com base no ID informado.
     *
     * @param id identificador único da marca a ser excluída
     */
    @Override
    public void excluir(Long id) {
        motoMarcaRepository.deleteById(id);
    }

    /**
     * Busca uma marca de motocicleta pelo seu ID.
     *
     * @param id identificador da marca
     * @return a marca correspondente
     * @throws IllegalArgumentException se o ID não existir
     */
    @Override
    @Transactional(readOnly = true)
    public MotoMarca buscarPorId(Long id) {
        return motoMarcaRepository.findById(id)
                .orElseThrow(() -> new IllegalArgumentException("Id Inválido para MotoMarca: " + id));
    }

    /**
     * Retorna todas as marcas de motocicleta cadastradas.
     *
     * @return lista de {@link MotoMarca}
     */
    @Override
    @Transactional(readOnly = true)
    public List<MotoMarca> buscarTodos() {
        return motoMarcaRepository.findAll();
    }

    /**
     * Busca e retorna uma lista de marcas de moto cujo nome contenha o texto
     * fornecido,
     * ignorando diferenças entre maiúsculas e minúsculas.
     *
     * <p>
     * A busca é feita de forma parcial, ou seja, marcas que contenham o trecho
     * informado
     * em qualquer posição do nome serão retornadas.
     * </p>
     *
     * @param nome o trecho do nome da marca a ser buscado.
     * @return uma lista de {@link MotoMarca} que contenham o nome informado.
     */
    @Override
    @Transactional(readOnly = true)
    public List<MotoMarca> buscarPorNomeContendo(String nome) {
        return motoMarcaRepository.findByNomeMarcaContainingIgnoreCase(nome.trim());
    }

    /**
     * Verifica se uma determinada marca possui modelos associados.
     *
     * @param id identificador da marca
     * @return true se houver pelo menos um modelo, false caso contrário
     */
    @Override
    public boolean motoMarcaTemModelos(Long id) {
        return !buscarPorId(id).getMotoModelo().isEmpty();
    }

    /**
     * Verifica se já existe uma marca com o nome informado.
     *
     * @param nomeMarca nome da marca a ser verificada
     * @return true se a marca existir, false caso contrário
     */
    @Override
    public boolean motoMarcaExiste(MotoMarca motoMarca) {
        motoMarca.setNomeMarca(StringUtil.capitalizarPrimeiraLetra(motoMarca.getNomeMarca()));
        return motoMarcaRepository.findByNomeMarca(motoMarca.getNomeMarca()) != null;
    }

    /**
     * Atualiza o status (ativo/inativo) de uma marca de motocicleta.
     *
     * @param id     identificador da marca
     * @param status novo valor de status (true para ativo, false para inativo)
     */
    @Override
    public void atualizarStatus(Long id, boolean status) {
        MotoMarca marca = buscarPorId(id);
        marca.setAtivo(status);
        motoMarcaRepository.save(marca);
    }

    /**
     * Realiza a busca paginada das marcas de moto, com ordenação conforme a direção
     * especificada.
     * 
     * <p>
     * Este método utiliza a API do JPA (`EntityManager`) para executar uma consulta
     * JPQL que retorna
     * uma lista de objetos {@link MotoMarca} de forma paginada, ou seja, limitada
     * ao número de registros
     * por página e deslocada de acordo com o número da página atual. A ordenação é
     * feita com base no nome da marca.
     * </p>
     * 
     * <p>
     * O tamanho fixo da página é definido como 5 registros. A página 1 representa
     * os primeiros 5 elementos.
     * A direção da ordenação pode ser "asc" (crescente) ou "desc" (decrescente).
     * </p>
     *
     * @param pagina  o número da página solicitada (baseado em 1, e não 0).
     * @param direcao a direção da ordenação, sendo "asc" para ascendente ou "desc"
     *                para descendente.
     * @return um objeto {@link PaginacaoUtil} contendo os dados da página atual
     *         (lista de marcas, total de páginas, etc.).
     */
    @Override
    public PaginacaoUtil<MotoMarca> buscaPaginada(int pagina, String direcao) {
        int tamanho = 5;
        int inicio = (pagina - 1) * tamanho; // Ex: página 1 começa do índice 0
        List<MotoMarca> marcas = entityManager
                .createQuery("select m from MotoMarca m order by m.nomeMarca " + direcao, MotoMarca.class)
                .setFirstResult(inicio)
                .setMaxResults(tamanho)
                .getResultList();

        long totalRegistros = count();
        long totalDePaginas = (totalRegistros + (tamanho - 1)) / tamanho;

        return new PaginacaoUtil<>(tamanho, pagina, totalDePaginas, direcao, marcas);
    }

    /**
     * Conta o número total de registros da entidade {@link MotoMarca} no banco de
     * dados.
     *
     * <p>
     * Utiliza uma consulta JPQL simples com a função de agregação
     * <code>count()</code> para retornar
     * o total de marcas de moto cadastradas.
     * </p>
     *
     * @return o número total de registros da entidade {@link MotoMarca}.
     */
    @Override
    public long count() {
        return entityManager
                .createQuery("select count(m) from MotoMarca m", Long.class)
                .getSingleResult();
    }

}
