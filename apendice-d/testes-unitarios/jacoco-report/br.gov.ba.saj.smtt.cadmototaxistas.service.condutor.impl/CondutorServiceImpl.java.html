package br.gov.ba.saj.smtt.cadmototaxistas.service.condutor.impl;

import java.time.LocalDate;
import java.util.Collections;
import java.util.List;
import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import br.gov.ba.saj.smtt.cadmototaxistas.auditoria.model.enums.AcaoAuditoria;
import br.gov.ba.saj.smtt.cadmototaxistas.auditoria.service.RegistroAuditoria.RegistroAuditoriaService;
import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Condutor;
import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.MotoModelo;
import br.gov.ba.saj.smtt.cadmototaxistas.repositories.CondutorRepository;
import br.gov.ba.saj.smtt.cadmototaxistas.service.condutor.CondutorService;
import br.gov.ba.saj.smtt.cadmototaxistas.service.motocicleta.modelo.MotoModeloService;
import br.gov.ba.saj.smtt.cadmototaxistas.util.paginacao.PaginacaoUtil;
import jakarta.persistence.EntityManager;
import jakarta.persistence.PersistenceContext;
import jakarta.persistence.TypedQuery;

/**
 * Implementação da interface {@link CondutorService}, responsável por fornecer
 * os serviços relacionados às entidades {@link Condutor}.
 *
 * Esta classe é anotada com {@code @Service} e {@code @Transactional} para
 * permitir
 * a gestão transacional e o controle do ciclo de vida dos métodos de serviço.
 */
@Service
@Transactional(readOnly = false)
public class CondutorServiceImpl implements CondutorService {

    @Autowired
    private CondutorRepository condutorRepository;

    @Autowired
    private MotoModeloService motoModeloService;

    @Autowired
    private RegistroAuditoriaService registroAuditoriaService;

    @PersistenceContext
    private EntityManager entityManager;

    /**
     * Salva um novo condutor, realizando a inicialização dos campos padrão,
     * gerando o número de ordem e registrando a auditoria da operação.
     * 
     * O método realiza duas operações de persistência:
     * 1. Salva inicialmente o condutor para obter o ID gerado.
     * 2. Atualiza o condutor com o número de ordem gerado a partir do ID.
     * 
     * @param condutor Condutor a ser salvo.
     */
    @Override
    public void salvar(Condutor condutor) {
        inicializarCamposPadrao(condutor);
        Condutor condutorSalvo = condutorRepository.save(condutor);
        gerarNumeroOrdem(condutorSalvo);

        condutorRepository.save(condutorSalvo);
        registrarAuditoria(AcaoAuditoria.CREATE, condutorSalvo);
    }

    /**
     * Edita um condutor existente, verificando se o alvará está vencido e
     * atualizando a data de vencimento se necessário.
     * 
     * Registra a auditoria da operação de atualização.
     * 
     * @param condutor Condutor a ser editado.
     */
    @Override
    public void editar(Condutor condutor) {
        Condutor condutorAntes = buscarPorId(condutor.getId());

        AcaoAuditoria acao;
        if (isAlvaraVencido(condutorAntes)) {
            condutor.setDataVencimentoAlvara(LocalDate.now().plusYears(1));
            acao = AcaoAuditoria.ATUALIZACAO_ALVARA;
        } else {
            acao = AcaoAuditoria.UPDATE;
        }

        condutorRepository.save(condutor);
        registrarAuditoria(acao, condutorAntes);
    }

    /**
     * Exclui um condutor pelo ID e registra auditoria da entidade excluída.
     */
    @Override
    public void excluir(Long id) {
        Condutor condutorAntes = buscarPorId(id);
        registrarAuditoria(AcaoAuditoria.DELETE, condutorAntes);
        condutorRepository.deleteById(id);
    }

    /**
     * Busca um condutor pelo ID.
     * 
     * @param id ID do condutor
     * @return Condutor encontrado ou lança IllegalArgumentException se não existir
     */
    @Override
    @Transactional(readOnly = true)
    public Condutor buscarPorId(Long id) {
        return condutorRepository.findById(id)
                .orElseThrow(() -> new IllegalArgumentException("Id Inválido para condutor: " + id));
    }

    /**
     * Busca todos os condutores, ordenados pelo número de ordem.
     * 
     * @return Lista de condutores ordenados
     */
    @Override
    @Transactional(readOnly = true)
    public List<Condutor> buscarTodos() {
        return condutorRepository.findAllByOrderByNumOrdemAsc();
    }

    /**
     * Busca um condutor pelo número de ordem.
     * 
     * @param numOrdem Número de ordem
     * @return Optional do condutor, se encontrado
     */
    @Override
    @Transactional(readOnly = true)
    public Optional<Condutor> buscarPorNumOrdem(String numOrdem) {
        return condutorRepository.findByNumOrdem(numOrdem);
    }

    /**
     * Busca um condutor pelo CPF.
     * 
     * @param cpf CPF do condutor
     * @return Condutor encontrado ou null se não existir
     */
    @Override
    @Transactional(readOnly = true)
    public Optional<Condutor> buscarPorCpf(String cpf) {
        return condutorRepository.findByCpf(cpf);
    }

    /**
     * Busca os modelos de motocicleta associados à marca selecionada no cadastro do
     * condutor.
     *
     * <p>
     * Este método extrai o ID da marca da motocicleta presente no objeto
     * {@code Condutor} e utiliza
     * o serviço {@code MotoModeloService} para recuperar todos os modelos dessa
     * marca.
     * </p>
     *
     * <p>
     * Se qualquer uma das informações necessárias estiver ausente (motocicleta,
     * modelo ou marca),
     * será lançada uma exceção informando que a marca não foi corretamente
     * informada.
     * </p>
     *
     * @param condutor o {@link Condutor} com os dados da motocicleta preenchidos
     * @return uma lista de {@link MotoModelo} vinculados à marca informada
     * @throws IllegalArgumentException se a marca da motocicleta não estiver
     *                                  informada corretamente
     */
    @Override
    public List<MotoModelo> buscarModelosDaMarcaSelecionada(Condutor condutor) {
        if (condutor.getMotocicleta() == null
                || condutor.getMotocicleta().getMotoModelo() == null
                || condutor.getMotocicleta().getMotoModelo().getMarcaMoto() == null
                || condutor.getMotocicleta().getMotoModelo().getMarcaMoto().getId() == null) {
            return Collections.emptyList();
        }

        Long marcaId = condutor.getMotocicleta()
                .getMotoModelo()
                .getMarcaMoto()
                .getId();

        return motoModeloService.findByMotoMarcaId(marcaId);
    }

    // Métodos para validação de existência, evitando duplicidade

    @Override
    public boolean existsByNumOrdemAndIdNot(String numOrdem, Long id) {
        return condutorRepository.existsByNumOrdemAndIdNot(numOrdem, id);
    }

    @Override
    public boolean existsByCpfAndIdNot(String cpf, Long id) {
        return condutorRepository.existsByCpfAndIdNot(cpf, id);
    }

    @Override
    public boolean existsByRgAndIdNot(String rg, Long id) {
        return condutorRepository.existsByRgAndIdNot(rg, id);
    }

    @Override
    public boolean existsByCnhNumeroAndIdNot(String cnh, Long id) {
        return condutorRepository.existsByCnhNumeroAndIdNot(cnh, id);
    }

    @Override
    public boolean existsByMotocicletaPlacaAndIdNot(String placa, Long id) {
        return condutorRepository.existsByMotocicletaPlacaAndIdNot(placa, id);
    }

    /**
     * Inicializa campos padrão em um novo condutor.
     * 
     * Se o condutor ainda não possui ID (ou seja, está sendo criado),
     * inicializa o número de ordem como null e define a data de vencimento do
     * alvará
     * para um ano a partir da data atual.
     * 
     * @param condutor Condutor que terá seus campos inicializados.
     */
    private void inicializarCamposPadrao(Condutor condutor) {
        if (condutor.getId() == null) {
            condutor.setNumOrdem(null);
            condutor.setDataVencimentoAlvara(LocalDate.now().plusYears(1));
        }
    }

    /**
     * Registra a ação de auditoria para o condutor.
     * 
     * @param acao     Tipo de ação realizada (CREATE, UPDATE, DELETE)
     * @param condutor Condutor relacionado à ação
     */
    private void registrarAuditoria(AcaoAuditoria acao, Condutor condutor) {
        registroAuditoriaService.registrar(acao, condutor);
    }

    /**
     * Atualiza o status de cadastro ativo/inativo de um condutor.
     * 
     * @param id     ID do condutor
     * @param status Novo status a ser definido
     */
    @Override
    @Transactional
    public void atualizarStatus(Long id, boolean status) {
        Condutor condutor = buscarPorId(id);
        condutor.setCadastroAtivo(status);
        condutorRepository.save(condutor);
    }

    /**
     * Realiza a busca paginada de condutores, podendo filtrar apenas aqueles com
     * alvará válido.
     *
     * @param pagina                número da página atual (a partir de 1)
     * @param direcao               direção da ordenação ("asc" ou "desc")
     * @param apenasComAlvaraValido define se deve buscar apenas condutores com
     *                              alvará válido
     * @return objeto {@link PaginacaoUtil} contendo os dados da página de
     *         condutores
     */
    @Override
    public PaginacaoUtil<Condutor> buscaPaginada(int pagina, String direcao, boolean apenasComAlvaraValido) {
        int tamanho = 5;
        int inicio = (pagina - 1) * tamanho;

        List<Condutor> condutores = buscarCondutoresPaginados(inicio, tamanho, direcao, apenasComAlvaraValido);
        long totalRegistros = contarCondutores(apenasComAlvaraValido);
        long totalDePaginas = (totalRegistros + (tamanho - 1)) / tamanho;

        return new PaginacaoUtil<>(tamanho, pagina, totalDePaginas, direcao, condutores);
    }

    /**
     * Executa a consulta paginada de condutores, com ou sem filtro por alvará
     * válido.
     *
     * @param inicio                índice inicial da consulta
     * @param tamanho               quantidade de registros por página
     * @param direcao               direção da ordenação ("asc" ou "desc")
     * @param apenasComAlvaraValido se true, busca apenas condutores com alvará
     *                              dentro da validade
     * @return lista de {@link Condutor} de acordo com os critérios
     */
    private List<Condutor> buscarCondutoresPaginados(int inicio, int tamanho, String direcao,
            boolean apenasComAlvaraValido) {
        String baseQuery = "from Condutor c";
        String where = apenasComAlvaraValido
                ? " where c.dataVencimentoAlvara is not null and c.dataVencimentoAlvara >= :hoje"
                : "";

        TypedQuery<Condutor> query = entityManager
                .createQuery("select c " + baseQuery + where + " order by c.numOrdem " + direcao, Condutor.class)
                .setFirstResult(inicio)
                .setMaxResults(tamanho);

        if (apenasComAlvaraValido) {
            query.setParameter("hoje", LocalDate.now());
        }

        return query.getResultList();
    }

    /**
     * Realiza a contagem total de condutores, podendo aplicar o filtro por alvará
     * válido.
     *
     * @param apenasComAlvaraValido se true, conta apenas condutores com alvará
     *                              dentro da validade
     * @return total de condutores conforme o critério
     */
    private long contarCondutores(boolean apenasComAlvaraValido) {
        String baseQuery = "from Condutor c";
        String where = apenasComAlvaraValido
                ? " where c.dataVencimentoAlvara is not null and c.dataVencimentoAlvara >= :hoje"
                : "";

        TypedQuery<Long> countQuery = entityManager
                .createQuery("select count(c) " + baseQuery + where, Long.class);

        if (apenasComAlvaraValido) {
            countQuery.setParameter("hoje", LocalDate.now());
        }

        return countQuery.getSingleResult();
    }

    /**
     * Gera e atribui o número de ordem ao condutor com base no seu ID e modalidade.
     * O formato do número de ordem será PREFIXO/NNNN, onde PREFIXO é obtido da
     * modalidade
     * do condutor e NNNN é o ID formatado com 4 dígitos com zeros à esquerda.
     * 
     * @param condutorSalvo Condutor já persistido no banco, com ID gerado.
     */
    private void gerarNumeroOrdem(Condutor condutorSalvo) {
        String prefixo = condutorSalvo.getModalidadeDeCondutor().getPrefixo(); // M, F, A
        String numeroFormatado = String.format("%04d", condutorSalvo.getId());
        condutorSalvo.setNumOrdem(prefixo + "/" + numeroFormatado);
    }

    /**
     * Verifica se o alvará do condutor está vencido.
     * 
     * @param condutor Condutor a ser verificado
     * @return true se o alvará estiver vencido, false caso contrário
     */
    private boolean isAlvaraVencido(Condutor condutor) {
        return Optional.ofNullable(condutor.getDataVencimentoAlvara())
                .map(data -> data.isBefore(LocalDate.now()))
                .orElse(false);
    }

}
