package br.gov.ba.saj.smtt.cadmototaxistas.service.condutor.validation.idademinima;

import java.time.LocalDate;

import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;

/**
 * Validador da anotação {@link ValidIdadeMinima}.
 * <p>
 * Valida se uma data de nascimento corresponde a uma idade mínima exigida.
 */
public class IdadeMinimaValidator implements ConstraintValidator<ValidIdadeMinima, LocalDate> {

  /**
   * Idade mínima exigida configurada na anotação.
   */
  private int idadeMinima;

  /**
   * Inicializa o validador com o valor da idade mínima definida na anotação.
   *
   * @param constraintAnnotation Anotação com o valor da idade mínima.
   */
  @Override
  public void initialize(ValidIdadeMinima constraintAnnotation) {
    this.idadeMinima = constraintAnnotation.value();
  }

  /**
   * Valida se a data de nascimento informada é anterior ou igual à data mínima calculada.
   * Se for nula, considera válida (validação opcional).
   *
   * @param dataNascimento Data de nascimento a ser validada.
   * @param context Contexto da validação.
   * @return {@code true} se a idade for maior ou igual à idade mínima, ou se nula;
   *         {@code false} caso contrário.
   */
  @Override
  public boolean isValid(LocalDate dataNascimento, ConstraintValidatorContext context) {
    if (dataNascimento == null) {
      return true;
    }
    LocalDate hoje = LocalDate.now();
    LocalDate dataMinima = hoje.minusYears(idadeMinima);
    return !dataNascimento.isAfter(dataMinima);
  }
}
