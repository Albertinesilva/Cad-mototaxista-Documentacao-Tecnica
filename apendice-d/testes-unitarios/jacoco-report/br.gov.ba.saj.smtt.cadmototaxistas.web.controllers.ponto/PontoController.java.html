package br.gov.ba.saj.smtt.cadmototaxistas.web.controllers.ponto;

import java.util.List;
import java.util.Optional;

import jakarta.validation.Valid;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.ui.ModelMap;
import org.springframework.validation.BindingResult;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.servlet.mvc.support.RedirectAttributes;

import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Ponto;
import br.gov.ba.saj.smtt.cadmototaxistas.service.ponto.PontoService;
import br.gov.ba.saj.smtt.cadmototaxistas.service.validacao.ValidadorDeContatoService;
import br.gov.ba.saj.smtt.cadmototaxistas.util.paginacao.PaginacaoUtil;

/**
 * Controller responsável por gerenciar as operações relacionadas aos pontos de
 * mototáxi.
 *
 * Fornece funcionalidades para cadastrar, listar com paginação, editar,
 * excluir, ativar/desativar
 * e buscar pontos por nome. Utiliza {@link PontoService} para executar a lógica
 * de negócio.
 */
@Controller
@RequestMapping("/pontos")
public class PontoController {

    private static final Logger LOGGER = LoggerFactory.getLogger(PontoController.class);

    @Autowired
    private PontoService pontoService;

    @Autowired
    private ValidadorDeContatoService validadorDeContatoService;

    /**
     * Exibe o formulário para cadastro de um novo ponto.
     *
     * @param ponto entidade ponto vazia
     * @return view da página de cadastro de ponto
     */
    @GetMapping("/cadastrar")
    public String cadastrar(Ponto ponto) {
        return "ponto/cadastro";
    }

    /**
     * Lista os pontos paginados e ordenados.
     *
     * @param model modelo da view
     * @param page  número da página atual (opcional)
     * @param dir   direção de ordenação (asc ou desc)
     * @return view com a lista paginada de pontos
     */
    @GetMapping("/listar")
    public String listar(ModelMap model, @RequestParam("page") Optional<Integer> page,
            @RequestParam("dir") Optional<String> dir) {
        int paginaAtual = page.orElse(1);
        String ordem = dir.orElse("asc");

        PaginacaoUtil<Ponto> pagePonto = pontoService.buscaPaginada(paginaAtual, ordem);
        model.addAttribute("pagePonto", pagePonto);
        return "ponto/lista";
    }

    /**
     * Salva um novo ponto após validação.
     *
     * @param ponto  entidade ponto a ser salva
     * @param result resultado da validação
     * @param model  modelo para erro
     * @param attr   atributos de redirecionamento
     * @return redirecionamento para listagem ou formulário em caso de erro
     */
    @PostMapping("/salvar")
    public String salvar(@Valid Ponto ponto, BindingResult result, Model model, RedirectAttributes attr) {
        if (result.hasErrors() || validadorDeContatoService.existeTelefoneCadastrado(ponto)) {
            model.addAttribute("ponto", ponto);
            model.addAttribute("fail",
                    "Erro ao salvar: verifique os campos obrigatórios e se o telefone informado já não está vinculado a outro contato.");
            LOGGER.error("Erro ao salvar ponto: {}", result.getAllErrors());
            return "ponto/cadastro";
        }
        pontoService.salvar(ponto);
        attr.addFlashAttribute("success", "Ponto inserido com sucesso.");
        return "redirect:/pontos/listar";
    }

    /**
     * Pré-carrega os dados do ponto a ser editado.
     *
     * @param id    identificador do ponto
     * @param model modelo com os dados
     * @return view de cadastro preenchida com os dados existentes
     */
    @GetMapping("/editar/{id}")
    public String preEditar(@PathVariable("id") Long id, ModelMap model) {
        model.addAttribute("ponto", pontoService.buscarPorId(id));
        return "ponto/cadastro";
    }

    /**
     * Edita um ponto após validação.
     *
     * @param ponto  entidade com dados atualizados
     * @param result resultado da validação
     * @param attr   atributos para mensagens flash
     * @return redirecionamento ou formulário em caso de erro
     */
    @PostMapping("/editar")
    public String editar(@Valid Ponto ponto, BindingResult result, Model model, RedirectAttributes attr) {
        if (result.hasErrors() || validadorDeContatoService.existeTelefoneCadastrado(ponto)) {
            model.addAttribute("fail",
                    "Erro ao salvar: verifique os campos obrigatórios e se o telefone informado já não está vinculado a outro contato.");
            LOGGER.error("Erro ao editar ponto: {}", result.getAllErrors());
            return "ponto/cadastro";
        }
        pontoService.editar(ponto);
        attr.addFlashAttribute("success", "Ponto editado com sucesso.");
        return "redirect:/pontos/listar";
    }

    /**
     * Exclui um ponto, caso não haja condutores vinculados.
     *
     * @param id   identificador do ponto
     * @param attr atributos para mensagens flash
     * @return redirecionamento para a listagem
     */
    @GetMapping("/excluir/{id}")
    public String excluir(@PathVariable("id") Long id, RedirectAttributes attr) {
        if (pontoService.existeCondutorCadastradoNoPonto(id)) {
            attr.addFlashAttribute("fail", "Não foi possível excluir o ponto, pois existe condutor(es) vinculado(s).");
        } else {
            pontoService.excluir(id);
            attr.addFlashAttribute("success", "Ponto removido com sucesso.");
        }
        return "redirect:/pontos/listar";
    }

    /**
     * Ativa ou desativa um ponto com base no status fornecido.
     *
     * @param id     identificador do ponto
     * @param status true para ativar, false para desativar
     * @param attr   atributos de redirecionamento
     * @return redirecionamento para listagem
     */
    @PostMapping("/ativar-desativar/{id}")
    public String ativarDesativar(@PathVariable Long id, @RequestParam boolean status, RedirectAttributes attr) {
        pontoService.atualizarStatus(id, status);
        if (status) {
            attr.addFlashAttribute("success", "Ponto ativado com sucesso.");
        } else {
            attr.addFlashAttribute("success", "Ponto desativado com sucesso.");
        }
        return "redirect:/pontos/listar";
    }

    @GetMapping("/buscar/nome")
    public String getPorNome(@RequestParam("nome") String nome, ModelMap model, RedirectAttributes attr) {
        List<Ponto> pontos = pontoService.buscarPorNomeContendo(nome);

        if (pontos.isEmpty()) {
            attr.addFlashAttribute("fail", "Nenhum ponto encontrado com esse nome.");
            return "redirect:/pontos/listar";
        }

        model.addAttribute("pontos", pontos);
        return "ponto/lista";
    }

}
