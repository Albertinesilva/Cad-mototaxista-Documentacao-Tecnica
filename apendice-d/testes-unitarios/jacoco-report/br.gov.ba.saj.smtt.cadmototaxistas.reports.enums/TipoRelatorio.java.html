package br.gov.ba.saj.smtt.cadmototaxistas.reports.enums;

/**
 * Enum que representa os diferentes tipos de relatórios disponíveis
 * para geração via JasperReports.
 *
 * <p>
 * Cada valor do enum contém o prefixo correspondente ao nome do arquivo
 * `.jasper` localizado na pasta <code>resources/reports</code>.
 * Esse prefixo é utilizado para compor dinamicamente o caminho do relatório.
 * </p>
 * 
 * Exemplos de nomes de arquivos gerados:
 * <ul>
 * <li><code>alvara-01.jasper</code></li>
 * <li><code>tipos-sanguineo-02.jasper</code></li>
 * </ul>
 * 
 * @author
 */
public enum TipoRelatorio {

  /**
   * Relatório de alvará de autorização para o condutor.
   */
  ALVARA_CONDUTOR("alvara-"),

  /**
   * Relatório estatístico com agrupamento por tipo sanguíneo dos condutores.
   */
  TIPOS_SANGUINEO("tipos-sanguineo-"),

  /**
   * Relatório de vencimento do alvará de autorização dos condutores.
   */
  VENCIMENTO_ALVARA("vencimento-alvara-"),

  /**
   * Relatório de motocicletas agrupadas por marca.
   */
  MARCAS_MOTOCICLETAS_MAIS_UTILIZADAS("motocicletas-mais-utilizadas-"),

  /**
   * Relatório de motocicletas agrupadas por ano de fabricação.
   */
  CONDUTORES_POR_FAIXA_ETARIA("condutores-faixa-etaria-"),

  /**
   * Relatório de motocicletas agrupadas por ano de fabricação.
   */
  CNH_VENCIDA_OU_PROXIMA_VENCIMENTO("cnh-situacao-");

  /**
   * Prefixo do nome do arquivo `.jasper`.
   */
  private final String nomeArquivo;

  /**
   * Construtor do enum.
   * 
   * @param nomeArquivo prefixo do nome do arquivo `.jasper` correspondente ao
   *                    tipo.
   */
  TipoRelatorio(String nomeArquivo) {
    this.nomeArquivo = nomeArquivo;
  }

  /**
   * Retorna o prefixo do nome do arquivo do relatório.
   * 
   * @return prefixo do arquivo Jasper correspondente.
   */
  public String getNomeArquivo() {
    return nomeArquivo;
  }

  /**
   * Converte uma string para o valor correspondente do enum
   * {@link TipoRelatorio},
   * ignorando diferenças entre maiúsculas e minúsculas.
   *
   * @param tipo nome textual do tipo de relatório (por exemplo:
   *             "ALVARA_CONDUTOR").
   * @return o valor correspondente do enum {@link TipoRelatorio}.
   * @throws IllegalArgumentException se o tipo informado não corresponder a
   *                                  nenhum valor do enum.
   */
  public static TipoRelatorio from(String tipo) {
    for (TipoRelatorio rel : values()) {
      if (rel.name().equalsIgnoreCase(tipo)) {
        return rel;
      }
    }
    throw new IllegalArgumentException("Relatório não encontrado: " + tipo);
  }
}
