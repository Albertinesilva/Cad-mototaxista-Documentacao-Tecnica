package br.gov.ba.saj.smtt.cadmototaxistas.security.services.email.impl;

import jakarta.mail.MessagingException;
import jakarta.mail.internet.MimeMessage;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.core.io.ClassPathResource;
import org.springframework.mail.javamail.JavaMailSender;
import org.springframework.mail.javamail.MimeMessageHelper;
import org.springframework.stereotype.Service;
import org.thymeleaf.context.Context;
import org.thymeleaf.spring6.SpringTemplateEngine;

import br.gov.ba.saj.smtt.cadmototaxistas.security.model.entities.Mail;
import br.gov.ba.saj.smtt.cadmototaxistas.security.services.email.EmailService;
import br.gov.ba.saj.smtt.cadmototaxistas.security.services.mail.MailService;

/**
 * Implementação do serviço de envio de e-mails para confirmação de cadastro
 * e redefinição de senha.
 */
@Service
public class EmailServiceImpl implements EmailService {

	@Value("${backend.url}")
	private String backendUrl;

	@Value("${frontend.url}")
	private String frontendUrl;

	@Autowired
	private MailService mailService;

	@Autowired
	private JavaMailSender mailSender;

	@Autowired
	private SpringTemplateEngine template;

	/**
	 * Envia um e-mail para o destinatário com um link para confirmação de cadastro.
	 * O e-mail utiliza template Thymeleaf para composição do conteúdo HTML.
	 *
	 * @param destino             endereço de e-mail do destinatário
	 * @param codigoDeConfirmacao código único para confirmação do cadastro
	 * @throws MessagingException caso ocorra erro no envio do e-mail
	 */
	@Override
	public void enviarPedidoDeConfirmacaoDeCadastro(String destino, String codigoDeConfirmacao)
			throws MessagingException {
		MimeMessage message = mailSender.createMimeMessage();
		MimeMessageHelper helper = new MimeMessageHelper(message, MimeMessageHelper.MULTIPART_MODE_MIXED_RELATED, "UTF-8");

		Context context = new Context();
		context.setVariable("titulo", "Bem vindo ao sistema de cadastro de mototaxistas, " + destino + "!");
		context.setVariable("texto", "Precisamos que confirme seu cadastro, clicando no link abaixo");
		context.setVariable("linkConfirmacao", backendUrl + "/u/confirmacao/cadastro?codigo=" + codigoDeConfirmacao);

		String html = template.process("email/confirmacao", context);
		helper.setTo(destino);
		helper.setText(html, true);
		helper.setSubject("Confirmacao de Cadastro");
		helper.setFrom("nao-responder@Smtt.com.br");
		helper.addInline("logo", new ClassPathResource("/static/image/logo-smtt-SEMF.jpeg"));

		mailSender.send(message);

		Mail mail = new Mail("smttp.gmail.com", destino, html, "Confirmacao de Cadastro");
		mailService.register(mail);
	}

	/**
	 * Envia um e-mail para o destinatário com o código de verificação para
	 * redefinição de senha.
	 * O e-mail utiliza template Thymeleaf para composição do conteúdo HTML.
	 *
	 * @param destino           endereço de e-mail do destinatário
	 * @param codigoVerificador código de verificação para validação da redefinição
	 * @throws MessagingException caso ocorra erro no envio do e-mail
	 */
	@Override
	public void enviarPedidoDeRedefinicaoSenha(String destino, String codigoVerificador) throws MessagingException {
		MimeMessage message = mailSender.createMimeMessage();
		MimeMessageHelper helper = new MimeMessageHelper(message, MimeMessageHelper.MULTIPART_MODE_MIXED_RELATED, "UTF-8");

		Context context = new Context();
		context.setVariable("titulo", "Redefinição de Senha");
		context.setVariable("texto", "Para redefinir sua senha use o código de verificação " +
				"quando exigido no formulário.");
		context.setVariable("verificador", codigoVerificador);

		String html = template.process(frontendUrl, context);
		helper.setTo(destino);
		helper.setText(html, true);
		helper.setSubject("Redefinição de Senha");
		helper.setFrom("no-replay@mototaxista.com.br");

		helper.addInline("logo", new ClassPathResource("/static/image/logo-smtt-SEMF.jpeg"));

		mailSender.send(message);

		Mail mail = new Mail("smttp.gmail.com", destino, html, "Redefinição de Senha");
		mailService.register(mail);
	}

}
