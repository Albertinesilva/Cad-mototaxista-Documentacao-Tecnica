package br.gov.ba.saj.smtt.cadmototaxistas.security.model.entities;

import java.time.Instant;

import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.AbstractEntity;
import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.FetchType;
import jakarta.persistence.JoinColumn;
import jakarta.persistence.ManyToOne;
import jakarta.persistence.Table;
import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * Representa um token de autenticação vinculado a um usuário.
 * <p>
 * Esse token é utilizado para operações como confirmação de cadastro e
 * redefinição de senha, tendo uma data de expiração e podendo ser desativado.
 * </p>
 * 
 * @author
 */
@AllArgsConstructor
@NoArgsConstructor
@Getter
@Setter
@Entity
@Table(name = "tokens")
public class Token extends AbstractEntity {

  /**
   * Código do token utilizado para autenticação ou confirmação.
   * <p>
   * Deve ser único e não nulo.
   */
  @Column(name = "token", nullable = false, unique = true, length = 255)
  private String token;

  /**
   * Instante exato da expiração do token.
   * <p>
   * Após essa data, o token é considerado inválido.
   */
  @Column(name = "data_expiracao", nullable = false)
  private Instant dataExpiracao;

  /**
   * Indica se o token foi desativado.
   * <p>
   * Tokens desativados não podem mais ser utilizados.
   */
  @Column(name = "desativado", nullable = false)
  private boolean desativado;

  /**
   * Usuário associado a este token.
   * <p>
   * Define a relação entre o token e o {@link Usuario} responsável.
   */
  @ManyToOne(fetch = FetchType.LAZY)
  @JoinColumn(name = "usuario_id")
  private Usuario usuario;

  /**
   * Método utilitário para desativar o token.
   */
  public void disabled() {
    this.desativado = true;
  }

}
