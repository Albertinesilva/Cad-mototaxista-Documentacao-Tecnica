package br.gov.ba.saj.smtt.cadmototaxistas.security.model.entities;

import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.Table;

import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.AbstractEntity;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * Representa o perfil de acesso de um usuário no sistema.
 * <p>
 * Cada perfil define um conjunto de permissões ou regras que um usuário pode
 * ter, como por exemplo: ADMIN, FUNCIONARIO etc.
 * <p>
 * A descrição do perfil deve ser única e não nula.
 *
 * @author
 */
@NoArgsConstructor
@Setter
@Getter
@Entity
@Table(name = "perfis")
public class Perfil extends AbstractEntity {

	/**
	 * Descrição do perfil de acesso (ex: ADMIN, FUNCIONARIO).
	 * <p>
	 * Deve ser única e não pode ser nula.
	 */
	@Column(name = "descricao", nullable = false, unique = true)
	private String descricao;

	/**
	 * Construtor que recebe o ID do perfil.
	 *
	 * @param id identificador do perfil.
	 */
	public Perfil(Long id) {
		super.setId(id);
	}

	/**
	 * Construtor que recebe apenas a descrição do perfil.
	 *
	 * @param descricao descrição textual do perfil.
	 */
	public Perfil(String descricao) {
		this.descricao = descricao;
	}

}
