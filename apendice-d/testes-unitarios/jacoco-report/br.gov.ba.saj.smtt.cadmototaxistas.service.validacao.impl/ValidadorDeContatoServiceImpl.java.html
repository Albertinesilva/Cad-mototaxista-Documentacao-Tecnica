package br.gov.ba.saj.smtt.cadmototaxistas.service.validacao.impl;

import java.util.stream.Stream;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Contato;
import br.gov.ba.saj.smtt.cadmototaxistas.model.interfaces.EntidadeComContato;
import br.gov.ba.saj.smtt.cadmototaxistas.repositories.ContatoRepository;
import br.gov.ba.saj.smtt.cadmototaxistas.service.validacao.ValidadorDeContatoService;

@Service
public class ValidadorDeContatoServiceImpl implements ValidadorDeContatoService {

  @Autowired
  private ContatoRepository contatoRepository;

  /**
   * Verifica se os telefones principal ou comercial informados no contato da
   * entidade já estão cadastrados no sistema.
   * 
   * <p>
   * Essa validação é útil para evitar registros duplicados de contatos com os
   * mesmos números de telefone.
   * Se o contato atual estiver sendo editado (ou seja, já possuir um ID), ele
   * será ignorado na verificação para
   * permitir atualização dos próprios dados sem bloquear por duplicidade.
   * </p>
   *
   * @param entidade a entidade que contém um {@link Contato}, como
   *                 {@code Cliente}, {@code Funcionario} ou {@code Condutor}.
   * @return {@code true} se existir outro contato com o mesmo telefone principal
   *         ou comercial (exceto o atual), {@code false} caso contrário.
   */
  @Override
  @Transactional(readOnly = true)
  public boolean existeTelefoneCadastrado(EntidadeComContato entidade) {
    Contato contato = entidade.getContato();
    Long idIgnorar = contato.getId();

    return Stream.of(contato.getTelefonePrincipal(), contato.getTelefoneComercial())
        .filter(telefone -> telefone != null && !telefone.isBlank())
        .flatMap(telefone -> contatoRepository
            .findByTelefonePrincipalOrTelefoneComercial(telefone, telefone)
            .stream())
        .anyMatch(c -> idIgnorar == null || !c.getId().equals(idIgnorar));
  }

}
