package br.gov.ba.saj.smtt.cadmototaxistas.service.motocicleta.validation.anomodelo;

import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Motocicleta;
import jakarta.validation.ConstraintValidator;
import jakarta.validation.ConstraintValidatorContext;

/**
 * Validador para a anotação {@link ValidAnoModelo}.
 * 
 * Realiza a validação da consistência entre os campos {@code anoFabricacao} e
 * {@code anoModelo}
 * da entidade {@link Motocicleta}, garantindo que o ano do modelo não seja
 * anterior ao ano de fabricação.
 */
public class AnoModeloValidator implements ConstraintValidator<ValidAnoModelo, Motocicleta> {

  /**
   * Realiza a validação do objeto {@link Motocicleta}.
   * 
   * @param moto    objeto a ser validado
   * @param context contexto da validação para adicionar mensagens personalizadas
   * @return {@code true} se válido, {@code false} caso contrário
   */
  @Override
  public boolean isValid(Motocicleta moto, ConstraintValidatorContext context) {
    if (moto == null) {
      return true; // Não valida se o objeto for nulo
    }

    if (moto.getAnoFabricacao() == null || moto.getAnoModelo() == null) {
      return true; // Não realiza validação se algum dos campos for nulo
    }

    boolean isValid = !moto.getAnoModelo().isBefore(moto.getAnoFabricacao());

    if (!isValid) {
      // Remove a mensagem de erro padrão e adiciona uma mensagem personalizada no
      // campo 'anoModelo'
      context.disableDefaultConstraintViolation();
      context.buildConstraintViolationWithTemplate(context.getDefaultConstraintMessageTemplate())
          .addPropertyNode("anoModelo")
          .addConstraintViolation();
    }

    return isValid;
  }
}
