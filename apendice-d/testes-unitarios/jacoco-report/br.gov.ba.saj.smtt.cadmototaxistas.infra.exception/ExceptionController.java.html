package br.gov.ba.saj.smtt.cadmototaxistas.infra.exception;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.dao.DataIntegrityViolationException;
import org.springframework.security.core.userdetails.UsernameNotFoundException;
import org.springframework.web.bind.annotation.ControllerAdvice;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.servlet.ModelAndView;

/**
 * Controlador global para tratamento de exceções.
 * 
 * Esta classe captura exceções lançadas durante a execução da aplicação e
 * direciona para páginas de erro amigáveis, com mensagens customizadas.
 * 
 * Trata exceções específicas como {@link UsernameNotFoundException} e
 * {@link DataIntegrityViolationException}, entre outras que podem ser
 * acrescentadas.
 * 
 */
@ControllerAdvice
public class ExceptionController {

	// Logger estático para registrar logs da classe
	private static final Logger LOGGER = LoggerFactory.getLogger(ExceptionController.class);

	/**
	 * Trata exceção quando usuário não é encontrado durante autenticação.
	 * 
	 * @param ex Exceção lançada quando o usuário não é encontrado.
	 * @return {@link ModelAndView} para a página de erro com informações do
	 *         problema.
	 */
	@ExceptionHandler(UsernameNotFoundException.class)
	public ModelAndView usuarioNaoEncontradoException(UsernameNotFoundException ex) {
		LOGGER.warn("UsernameNotFoundException: {}", ex.getMessage());
		ModelAndView model = new ModelAndView("error");
		model.addObject("status", 404);
		model.addObject("error", "Operação não pode ser realizada.");
		model.addObject("message", ex.getMessage());
		return model;
	}

	/**
	 * Trata exceções de violação de integridade de dados no banco, como
	 * valores duplicados em campos únicos.
	 * 
	 * @param ex Exceção lançada em violação de restrições do banco de dados.
	 * @return {@link ModelAndView} para a página de erro com mensagens específicas.
	 */
	@ExceptionHandler(DataIntegrityViolationException.class)
	public ModelAndView dataIntegrityViolationException(DataIntegrityViolationException ex) {
		LOGGER.error("DataIntegrityViolationException: ", ex);
		ModelAndView model = new ModelAndView("error");
		model.addObject("status", 400);
		model.addObject("error", "Violação de Integridade de Dados");

		String mensagem = "Operação não pode ser realizada devido à integridade dos dados.";

		if (ex.getMessage().contains("condutor_numero_ordem_key")) {
			mensagem = "Número de Ordem já cadastrado.";
		} else if (ex.getMessage().contains("condutor_cpf_key")) {
			mensagem = "CPF já cadastrado.";
		} else if (ex.getMessage().contains("condutor_rg_key")) {
			mensagem = "RG já cadastrado.";
		}

		model.addObject("message", mensagem);
		return model;
	}

	/**
	 * Trata exceção genérica para erros inesperados do servidor.
	 * 
	 * @param ex Exceção genérica lançada.
	 * @return {@link ModelAndView} para página de erro 500.
	 */
	@ExceptionHandler(Exception.class)
	public ModelAndView handleGenericException(Exception ex) {
		LOGGER.error("Erro inesperado: ", ex);
		ModelAndView model = new ModelAndView("error");
		model.addObject("status", 500);
		model.addObject("error", "Erro Interno do Servidor");
		model.addObject("message", "Ocorreu um erro inesperado. Por favor, tente novamente mais tarde.");
		return model;
	}

	@ExceptionHandler(AcessoNegadoException.class)
	public ModelAndView acessoNegadoException(AcessoNegadoException ex) {
		// Criação da ModelAndView para exibir a página de erro
		ModelAndView mv = new ModelAndView("error");

		// Adicionando atributos à ModelAndView para exibição
		mv.addObject("status", 403);
		mv.addObject("error", "Operação não pode ser realizada");
		mv.addObject("message", ex.getMessage());

		return mv;
	}

}
