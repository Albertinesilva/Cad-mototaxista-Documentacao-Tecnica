package br.gov.ba.saj.smtt.cadmototaxistas.security.services.token.impl;

import java.security.SecureRandom;
import java.time.Instant;
import java.time.LocalDateTime;
import java.time.ZoneOffset;
import java.util.Base64;
import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import br.gov.ba.saj.smtt.cadmototaxistas.security.model.entities.Token;
import br.gov.ba.saj.smtt.cadmototaxistas.security.model.entities.Usuario;
import br.gov.ba.saj.smtt.cadmototaxistas.security.model.enums.SecureTokenType;
import br.gov.ba.saj.smtt.cadmototaxistas.security.repositories.TokenRepository;
import br.gov.ba.saj.smtt.cadmototaxistas.security.services.token.TokenService;

/**
 * Implementação dos serviços relacionados à geração, persistência e busca de tokens de segurança.
 */
@Service
public class TokenServiceImpl implements TokenService {

  private static final SecureRandom SECURE_RANDOM = new SecureRandom();

  @Autowired
  private TokenRepository tokenRepository;

  /**
   * Salva um token no repositório.
   * 
   * @param token token a ser salvo.
   */
  @Override
  public void salvar(Token token) {
    tokenRepository.save(token);
  }

  /**
   * Gera um token de ativação codificado em Base64 para o usuário informado.
   * 
   * @param usuario usuário para quem o token será gerado.
   * @param type tipo do token de segurança.
   * @return token de ativação codificado em Base64.
   */
  @Override
  public String generateTokenAtivacao(Usuario usuario) {
    byte[] randomBytes = new byte[SecureTokenType.TOKEN_ATIVACAO.getByteLength()];
    SECURE_RANDOM.nextBytes(randomBytes);
    String tokenAtivacao = Base64.getUrlEncoder().withoutPadding().encodeToString(randomBytes);

    Instant dataExpiracao = expiracaoDataAtivacao();
    Token tokenObj = new Token(tokenAtivacao, dataExpiracao, false, usuario);
    salvar(tokenObj);

    return tokenObj.getToken();
  }

  /**
   * Gera um token para redefinição de senha do usuário informado, salva no banco e retorna o objeto Token.
   * 
   * @param usuario usuário para quem o token será gerado.
   * @return token para redefinição de senha.
   */
  @Override
  public Token generateTokenParaRedefinicaoSenha(Usuario usuario) {
    byte[] randomBytes = new byte[SecureTokenType.TOKEN_REDEFINICAO_SENHA.getByteLength()];
    SECURE_RANDOM.nextBytes(randomBytes);
    String tokenRedefinicao = Base64.getUrlEncoder().withoutPadding().encodeToString(randomBytes);

    Instant dataExpiracao = expiracaoDataRedefinicaoSenha();
    Token tokenObj = new Token(tokenRedefinicao, dataExpiracao, false, usuario);
    salvar(tokenObj);

    return tokenObj;
  }

  /**
   * Gera um código verificador codificado em Base64.
   * 
   * @return código verificador codificado em Base64.
   */
  @Override
  public String generateCodigoVerificador() {
    byte[] randomBytes = new byte[SecureTokenType.CODIGO_VERIFICACAO.getByteLength()];
    SECURE_RANDOM.nextBytes(randomBytes);
    return Base64.getUrlEncoder().withoutPadding().encodeToString(randomBytes);
  }

  /**
   * Obtém a data de expiração do token de ativação.
   * 
   * @return data de expiração do token de ativação (2 horas a partir do momento atual).
   */
  private Instant expiracaoDataAtivacao() {
    return LocalDateTime.now().plusHours(2).toInstant(ZoneOffset.of("-03:00"));
  }

  /**
   * Obtém a data de expiração do token de redefinição de senha.
   * 
   * @return data de expiração do token de redefinição de senha (30 minutos a partir do momento atual).
   */
  private Instant expiracaoDataRedefinicaoSenha() {
    return LocalDateTime.now().plusMinutes(30).toInstant(ZoneOffset.of("-03:00"));
  }

  /**
   * Busca um token pelo seu valor.
   * 
   * @param token valor do token.
   * @return Optional contendo o token, caso encontrado.
   */
  @Override
  public Optional<Token> findByToken(String token) {
    return tokenRepository.findByToken(token);
  }
}
