package br.gov.ba.saj.smtt.cadmototaxistas.datatables;

/**
 * Classe utilitária que define as colunas a serem utilizadas nas configurações
 * de ordenação e exibição dos componentes DataTables da aplicação.
 *
 * <p>
 * Cada constante representa as colunas mapeadas para determinada entidade
 * (por exemplo, {@code Funcionario}, {@code Usuario}), sendo usadas para fins de
 * paginação, ordenação e exibição no front-end.
 * </p>
 *
 * <p>
 * Essas colunas devem estar alinhadas com os atributos do objeto exibido
 * na tabela HTML para que o plugin DataTables funcione corretamente.
 * </p>
 *
 * @author SeuNome
 * @since 1.0
 */
public class DatatablesColunas {

    /**
     * Colunas utilizadas pelo DataTables para exibição e ordenação da entidade {@code Funcionario}.
     * <ul>
     *   <li>id - Identificador único do funcionário</li>
     *   <li>nome - Nome completo do funcionário</li>
     *   <li>dtInscricao - Data de inscrição no sistema</li>
     * </ul>
     */
    public static final String[] FUNCIONARIO = {"id", "nome", "dtInscricao"};

    /**
     * Colunas utilizadas pelo DataTables para exibição e ordenação da entidade {@code Usuario}.
     * <ul>
     *   <li>id - Identificador único do usuário</li>
     *   <li>email - Endereço de e-mail do usuário</li>
     *   <li>ativo - Status de ativação do usuário</li>
     *   <li>perfis - Perfis ou permissões atribuídas ao usuário</li>
     * </ul>
     */
    public static final String[] USUARIOS = {"id", "email", "ativo", "perfis"};
}
