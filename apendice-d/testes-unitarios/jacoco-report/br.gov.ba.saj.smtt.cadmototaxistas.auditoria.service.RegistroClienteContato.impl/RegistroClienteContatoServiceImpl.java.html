package br.gov.ba.saj.smtt.cadmototaxistas.auditoria.service.RegistroClienteContato.impl;

import java.time.LocalDateTime;
import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import br.gov.ba.saj.smtt.cadmototaxistas.auditoria.model.entities.RegistroClienteContato;
import br.gov.ba.saj.smtt.cadmototaxistas.auditoria.repository.RegistroClienteContatoRepository;
import br.gov.ba.saj.smtt.cadmototaxistas.auditoria.service.RegistroClienteContato.RegistroClienteContatoService;
import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Cliente;
import br.gov.ba.saj.smtt.cadmototaxistas.model.entities.Condutor;
import br.gov.ba.saj.smtt.cadmototaxistas.service.cliente.ClienteService;
import br.gov.ba.saj.smtt.cadmototaxistas.service.condutor.CondutorService;
import br.gov.ba.saj.smtt.cadmototaxistas.util.paginacao.PaginacaoUtil;
import jakarta.persistence.EntityManager;
import jakarta.persistence.EntityNotFoundException;
import jakarta.persistence.PersistenceContext;

@Service
public class RegistroClienteContatoServiceImpl implements RegistroClienteContatoService {

  @Autowired
  private RegistroClienteContatoRepository registroClienteContatoRepository;

  @Autowired
  private CondutorService condutorService;

  @Autowired
  private ClienteService clienteService;

  @PersistenceContext
  private EntityManager entityManager;

  @Override
  @Transactional(readOnly = false)
  public void salvar(Long condutorId, String emailClienteAutenticado) {
    Condutor condutor = condutorService.buscarPorId(condutorId);
    Cliente cliente = clienteService.buscarPorUsuarioEmail(emailClienteAutenticado)
        .orElseThrow(
            () -> new EntityNotFoundException("Cliente não encontrado com o email: " + emailClienteAutenticado));

    var registro = new RegistroClienteContato();
    registro.setCliente(cliente);
    registro.setNomeCliente(cliente.getNome());
    registro.setEmailCliente(cliente.getUsuario().getEmail());
    registro.setNomeCondutor(condutor.getNome());
    registro.setNumeroOrdem(condutor.getNumOrdem());
    registro.setContatoCondutor(condutor.getContato().getTelefoneComercial());
    registro.setMarcaModeloMoto(
        condutor.getMotocicleta().getMotoModelo().getMarcaMoto().getNomeMarca()
            + " - "
            + condutor.getMotocicleta().getMotoModelo().getNomeModelo());
    registro.setDataContato(LocalDateTime.now());
    registroClienteContatoRepository.save(registro);

  }

  @Override
  @Transactional(readOnly = true)
  public List<RegistroClienteContato> listarPorEmailCliente(String email) {
    return registroClienteContatoRepository.findByClienteUsuarioEmail(email);
  }

  /**
   * Realiza a busca paginada dos registros de contatos de cliente, com ordenação
   * conforme a direção
   * especificada.
   * 
   * <p>
   * Este método utiliza a API do JPA (`EntityManager`) para executar uma consulta
   * JPQL que retorna
   * uma lista de objetos {@link RegistroClienteContato} de forma paginada, ou
   * seja, limitada
   * ao número de registros
   * por página e deslocada de acordo com o número da página atual. A ordenação é
   * feita com base no número de ordem do condutor.
   * </p>
   * 
   * <p>
   * O tamanho fixo da página é definido como 5 registros. A página 1 representa
   * os primeiros 5 elementos.
   * A direção da ordenação pode ser "asc" (crescente) ou "desc" (decrescente).
   * </p>
   *
   * @param pagina  o número da página solicitada (baseado em 1, e não 0).
   * @param direcao a direção da ordenação, sendo "asc" para ascendente ou "desc"
   *                para descendente.
   * @return um objeto {@link PaginacaoUtil} contendo os dados da página atual
   *         (lista de registos de contatos de cliente, total de páginas, etc.).
   */
  @Override
  public PaginacaoUtil<RegistroClienteContato> buscaPaginada(Long clienteId, int pagina, String direcao) {
    int tamanho = 5;
    int inicio = (pagina - 1) * tamanho;

    List<RegistroClienteContato> registroClienteContatos = entityManager
        .createQuery(
            "select r from RegistroClienteContato r where r.cliente.id = :clienteId order by r.numeroOrdem " + direcao,
            RegistroClienteContato.class)
        .setParameter("clienteId", clienteId)
        .setFirstResult(inicio)
        .setMaxResults(tamanho)
        .getResultList();

    long totalRegistros = count(clienteId);
    long totalDePaginas = (totalRegistros + (tamanho - 1)) / tamanho;

    return new PaginacaoUtil<>(tamanho, pagina, totalDePaginas, direcao, registroClienteContatos);
  }

  /**
   * Conta o número total de registros da entidade {@link RegistroClienteContato}
   * no banco de
   * dados.
   *
   * <p>
   * Utiliza uma consulta JPQL simples com a função de agregação
   * <code>count()</code> para retornar
   * o total de registros existentes na tabela
   * <code>registro_cliente_contato</code>.
   * </p>
   *
   * @return o número total de registros da entidade
   *         {@link RegistroClienteContato}.
   */
  @Override
  public long count(Long clienteId) {
    return entityManager.createQuery(
        "select count(r) from RegistroClienteContato r where r.cliente.id = :clienteId", Long.class)
        .setParameter("clienteId", clienteId)
        .getSingleResult();
  }

}
