package br.gov.ba.saj.smtt.cadmototaxistas.auditoria.model.enums;

/**
 * Enumeração que representa os tipos de ações que podem ser auditadas no
 * sistema.
 *
 * Cada constante define uma ação executada sobre os dados (como criação ou
 * exclusão),
 * junto de uma descrição legível que pode ser exibida em interfaces ou
 * relatórios.
 * 
 * Usado em conjunto com
 * {@link br.gov.ba.saj.smtt.cadmototaxistas.auditoria.model.entities.RegistroAuditoria}
 * para registrar e categorizar eventos do sistema.
 * 
 * @author SeuNome
 * @since 1.0
 */
public enum AcaoAuditoria {

    /**
     * Ação de criação de um novo registro no sistema.
     */
    CREATE("Criar"),

    /**
     * Ação de modificação/atualização de um registro existente.
     */
    UPDATE("Editar"),

    /**
     * Ação de remoção/exclusão de um registro do sistema.
     */
    DELETE("Excluir"),

    /**
     * Ação de consulta ou visualização de um dado.
     * Pode ser utilizada em auditorias de acessos ou verificações.
     */
    CONSULT("Consultar"),

    ATUALIZACAO_ALVARA("Atualização de Alvará");


    /**
     * Descrição legível da ação, usada para exibição em logs e interfaces.
     */
    private final String descricao;

    /**
     * Construtor do enum que associa uma descrição à ação.
     * 
     * @param descricao Descrição legível da ação.
     */
    AcaoAuditoria(String descricao) {
        this.descricao = descricao;
    }

    /**
     * Retorna a descrição legível da ação de auditoria.
     * 
     * @return descrição da ação.
     */
    public String getDescricao() {
        return descricao;
    }
}
