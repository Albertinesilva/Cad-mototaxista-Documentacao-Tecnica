package br.gov.ba.saj.smtt.cadmototaxistas.util.string;

import java.util.Arrays;
import java.util.HashSet;
import java.util.Set;

/**
 * Classe utilitária para manipulação de textos e strings no sistema
 * CADMOTOTAXISTA.
 */
public class StringUtil {

  /**
   * Capitaliza apenas a primeira letra da string e converte o restante para
   * minúsculas.
   * <p>
   * Exemplo: "yamaha" → "Yamaha"
   *
   * @param texto texto a ser formatado
   * @return texto formatado com a primeira letra maiúscula
   */
  public static String capitalizarPrimeiraLetra(String texto) {
    if (texto == null || texto.isBlank()) {
      return texto;
    }

    texto = texto.trim();
    return texto.substring(0, 1).toUpperCase() + texto.substring(1).toLowerCase();
  }

  /**
   * Formata um nome próprio capitalizando a primeira letra de cada palavra,
   * exceto preposições e artigos comuns da língua portuguesa, que permanecem em
   * minúsculo,
   * a menos que estejam no início do nome.
   * 
   * Exemplo:
   * 
   * <pre>
   * "joão da silva" -> "João da Silva"
   * "ana maria de souza" -> "Ana Maria de Souza"
   * </pre>
   * 
   * @param input a string com o nome a ser formatado
   * @return o nome formatado com a capitalização correta, ou a mesma string se
   *         for nula ou vazia
   */
  public static String capitalizeName(String input) {
    if (input == null || input.isBlank())
      return input;

    String[] lowercaseWords = { "de", "da", "do", "das", "dos", "e", "em", "no", "na", "nos", "nas", "por", "para" };
    Set<String> exceptions = new HashSet<>(Arrays.asList(lowercaseWords));

    String[] words = input.toLowerCase().split("\\s+");
    StringBuilder result = new StringBuilder();

    for (int i = 0; i < words.length; i++) {
      String word = words[i];
      if (i != 0 && exceptions.contains(word)) {
        result.append(word);
      } else {
        // Capitaliza a primeira letra e concatena o restante em minúsculo
        result.append(word.substring(0, 1).toUpperCase());
        if (word.length() > 1) {
          result.append(word.substring(1));
        }
      }
      if (i < words.length - 1) {
        result.append(" ");
      }
    }

    return result.toString();
  }

}
